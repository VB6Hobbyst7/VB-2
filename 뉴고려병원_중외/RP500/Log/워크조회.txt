  FROM SLA_LabMaster 
 WHERE RECEIPTDATE between '2017-05-01' and '2017-05-01'
   AND LABCODE IN ('C3815N1','C3815N2','C3815N3','C3815N4','C3815N5','C3815N6','C3815N7','C3815N8','L3515N9') 
   AND JSTATUS < '3'
  ORDER BY RECEIPTDATE
