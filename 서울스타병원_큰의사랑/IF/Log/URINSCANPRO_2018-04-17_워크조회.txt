Select DISTINCT ORDDATE as HOSPDATE, SPCID as BARCODE , PATNO as PID , WORKCODE as CHARTNO   From SLXWORKT  Where ORDDATE BETWEEN  '20180417' And '20180417'   And HOSPID   = 'I200'   And ROOMCODE = 'LU'   And EXAMCODE IN ('B0030A','B0030I','B0030B','B0030C','B0030D','B0030E','B0030F','B0030G','B0030J','B0030H')   And (RSLTTEXT = '' or RSLTTEXT is null)  ORDER BY ORDDATE, SPCID, PATNO, WORKCODE, ORDSEQNO
