SELECT DISTINCT        READING_YMD                 AS HOSPDATE                 
     , BCODE_NO                    AS BARCODE                  
     , PTNT_NO                     AS PID                      
     , PTNT_NM                     AS PNAME                    
     , AGE                         AS AGE                      
     , SEX                         AS SEX                      
     , IO_GB                       AS INOUT                    
     , COUNT(ORD_CD)               AS COUNT                    
  FROM LIS_INTERFACE1_V                                        
 WHERE READING_YMD BETWEEN '20200611' AND '20200611'   
   AND ORD_CD IN ('LU10101','LU10103','LU10104','LU10105','LU10106','LU10107','LU10108')                          
   AND STS_CD = '0'                                            
 GROUP BY READING_YMD,BCODE_NO,PTNT_NO,PTNT_NM,AGE,SEX,IO_GB   
 ORDER BY READING_YMD,PTNT_NO,BCODE_NO
