SELECT DISTINCT        READING_YMD     AS HOSPDATE         
     , BCODE_NO        AS BARCODE          
     , PTNT_NO         AS PID              
     , PTNT_NM         AS PNAME            
     , AGE             AS AGE              
     , SEX             AS SEX              
     , IO_GB           AS INOUT            
     , ORD_CD          AS ITEM             
     , SP_CD           AS SPCCD            
  FROM LIS_INTERFACE1_V                    
 WHERE BCODE_NO    = '100185'  
   AND READING_YMD = '2019-10-01'  
   AND STS_CD = '0'                        
   AND ORD_CD IN ('LU10101','LU10102','LU10103','LU10104','LU10105','LU10106','LU10107','LU10108','LU10109','LU10110','LU10111')      
 ORDER BY ORD_CD
