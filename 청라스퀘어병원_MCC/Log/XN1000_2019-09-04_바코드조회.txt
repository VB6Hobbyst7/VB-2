SELECT DISTINCT        READING_YMD     AS HOSPDATE         
     , BCODE_NO        AS BARCODE          
     , PTNT_NO         AS PID              
     , PTNT_NM         AS PNAME            
     , AGE             AS AGE              
     , SEX             AS SEX              
     , IO_GB           AS INOUT            
     , ORD_CD          AS ITEM             
     , SP_CD           AS SPCCD            
  FROM LIS_INTERFACE1_V                    
 WHERE BCODE_NO    = '1000001207'  
   AND READING_YMD = '2019-09-04'  
   AND STS_CD = '0'                        
   AND ORD_CD IN ('LB1050','LB1040','LB1010','LB1020','LB218','LB219','LB220','LB1060','LH10207','LH10208','LH10201','LH10204','LH10205','LH124','LH123','LB1220','LH116')      
 ORDER BY ORD_CD
