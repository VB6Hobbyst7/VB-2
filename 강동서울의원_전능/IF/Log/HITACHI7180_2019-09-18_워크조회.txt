SELECT DISTINCT        READING_YMD                 AS HOSPDATE                 
     , BCODE_NO                    AS BARCODE                  
     , PTNT_NO                     AS PID                      
     , PTNT_NM                     AS PNAME                    
     , AGE                         AS AGE                      
     , SEX                         AS SEX                      
     , IO_GB                       AS INOUT                    
     , COUNT(ORD_CD)               AS COUNT                    
  FROM LIS_INTERFACE1_V                                        
 WHERE READING_YMD BETWEEN '20190918' AND '20190918'   
   AND ORD_CD IN ('LC3730','LC3750','LC3721','LC3720','LB2602','LB2611','LC2200','LC2210','LC2420','LC2443P','LC2443','LC3711P','LC3711','LB2710','LB2590','LC3780','LC3795','LC3794','LB2570','LB2580','LC2411','LC2243','LC2241','LC4903','LC4901','LC2430','LCZ133','LCZ241','LCZ241','LCZ2246','C2430A','eGFR','LCZ241')                          
   AND STS_CD = '0'                                            
 GROUP BY READING_YMD,BCODE_NO,PTNT_NO,PTNT_NM,AGE,SEX,IO_GB   
 ORDER BY READING_YMD,PTNT_NO,BCODE_NO
