  FROM LIS_INTERFACE1_V

   AND BCODE_NO    = '100115'  
   AND ORD_CD IN ('LU10101','LU10102','LU10103','LU10104','LU10105','LU10106','LU10107','LU10108','LU10109','LU10110','LU10111')
