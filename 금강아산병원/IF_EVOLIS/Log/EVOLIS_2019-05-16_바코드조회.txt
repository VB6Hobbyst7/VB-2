SELECT DISTINCT OKDT  as HOSPDATE, SPNO as BARCODE, WKNO as CHARTNO, PAID as PID, PANM as PNAME, SEXS as SEX, ORCD as ITEM 
  FROM V_AFISINFO 
 WHERE OKFL = 'N'
   AND SPNO = 'S4' 
   AND ORCD IN ('L435051','L435051_C','L43509','L43509_C','L43171','L43171_C','L9018','L901801','L901802','L901803')
   AND HOCD = '60'
 ORDER BY OKDT
