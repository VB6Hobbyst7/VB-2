SELECT DISTINCT OKDT  as HOSPDATE, SPNO as BARCODE, WKNO as CHARTNO, PAID as PID, PANM as PNAME, SEXS as SEX, ORCD as ITEM 
  FROM V_AFISINFO 
 WHERE OKFL = 'N'
   AND SPNO = '190513400075' 
   AND ORCD IN ('L435051','L43509','L43171','L9018','L901801','L901802','L901803')
   AND HOCD = '60'
 ORDER BY OKDT
