SELECT DISTINCT OKDT  as HOSPDATE, SPNO as BARCODE, WKNO as CHARTNO, PAID as PID, PANM as PNAME, SEXS as SEX  FROM V_AFISINFO  WHERE OKFL = 'N'   AND OKDT BETWEEN  TO_DATE('20190716','yyyymmdd') AND TO_DATE('20190716','yyyymmdd') + 1   AND ORCD IN ('L435051','L43171','L43509','L9018','LXXXXX','LYYYY','LZZZZ','L901810','L901811','L901812')   AND HOCD = '60' ORDER BY OKDT
