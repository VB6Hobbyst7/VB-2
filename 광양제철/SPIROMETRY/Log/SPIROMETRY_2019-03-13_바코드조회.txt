SELECT DISTINCT       a.per_gumjin_date   AS HOSPDATE                                  , a.per_name          AS PNAME                                     , a.per_jikbun        AS BARCODE                                   , b.meditem           AS ITEM                                   FROM TB_JUPSU a                                                       , TB_JUPSU_ITEM b                                                  , BAG_CODEVALUE c                                              WHERE a.com_code = '1'                                               AND a.per_jikbun            = '370942'                 AND a.per_ssn               = b.per_ssn                            AND a.per_jupsu_date        = b.per_jupsu_date                     AND a.per_ssn               = b.per_ssn                            AND a.per_gumjin_date       = b.per_gumjin_date                    AND b.meditem               IN ( '000006000','000006003','000006001','N/A','N/A','000006004','N/A','N/A','N/A','000006006','N/A','N/A','N/A','N/A','N/A','N/A','N/A','000006012','N/A','N/A','N/A','N/A','N/A','N/A','N/A','N/A','N/A')               AND (b.RESULT = '' OR b.RESULT IS NULL)                             AND 'P00' || a.jupsu_gubun  = c.gumsacode                          AND c.codegubun = 'P00'                                          ORDER BY b.meditem
