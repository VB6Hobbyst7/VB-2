SELECT DISTINCT        a.RECEIPTDATE    AS HOSPDATE      , a.SPECIMENNUM    AS BARCODE       , a.RECEIPTNO      AS CHARTNO       , a.IPDOPD         AS INOUT         , a.PTNO           AS PID           , a.SNAME          AS PNAME         , a.ORDERCODE      AS ORDCODE       , b.LABCODE        AS ITEM        FROM SLA_LabMaster a, SLA_LabResult b    WHERE a.RECEIPTNO     = b.RECEIPTNO          AND a.ORDERCODE     = b.ORDERCODE          and a.RECEIPTDATE   = b.RECEIPTDATE        AND a.SPECIMENNUM   = b.SPECIMENNUM        AND a.SPECIMENNUM   = '00328354'   AND b.LABCODE IN ('D0002010','L2011Z','D0002030','L2012Z','D0002050','L2013Z','D0002040','L2014Z','D0002070','L2015Z','L201916','L201917','L201918','L201919','L201920','L2081','L2082','L2083','L2021','L2027','L2022')      AND a.JSTATUS < '3'                      ORDER BY b.LABCODE
