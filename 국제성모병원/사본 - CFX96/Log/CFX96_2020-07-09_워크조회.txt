SELECT DISTINCT   JUBSU_DATE       AS HOSPDATE         
, SPECIMEN_SER     AS BARCODE          
, BUNHO            AS PID              
, SUNAME           AS PNAME            
, AGE              AS AGE              
, SEX              AS SEX              
, ORDER_INPUT      AS DEPTNM           
, COUNT(SUB_HANGMOG_CODE) as CNT       
  FROM VW_IHIS_LISDIE_GUMSA_INFO_SUB   
 WHERE JUBSU_DATE BETWEEN '20200709' AND '20200709' 
   AND SUB_HANGMOG_CODE IN ('ZZZZZ','YYYYY','XXXXX','LPD336','LPD331')
 GROUP BY JUBSU_DATE,SPECIMEN_SER,BUNHO,SUNAME,AGE,SEX,ORDER_INPUT  
 ORDER BY JUBSU_DATE,BUNHO
