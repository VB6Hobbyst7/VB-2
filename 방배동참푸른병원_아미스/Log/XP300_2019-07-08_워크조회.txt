SELECT DISTINCT       O.ACPTDATE              as HOSPDATE                 
     , R.SPCMNO                as BARCODE                  
     , P.PATID                 as PID                      
     , P.PATNAME               as PNAME                    
     , P.SEX                   as SEX                      
     , COUNT(R.RESULTITEMCODE) as CNT                      
  FROM REGISTINFOS O, RESULTOFNUM R, PATMST P              
 WHERE O.ACPTDATE  = R.ACPTDATE                            
   AND O.PATID     = R.PATID                               
   AND O.ACPTSEQ   = R.ACPTSEQ                             
   AND O.PATID     = P.PATID                               
   AND O.ACPTDATE BETWEEN '20190708' and '20190708'
   AND R.RESULTITEMCODE IN ('D0002010','D0002030','D0002050','D0002040','XH002005','XH002006','XH002007','D0002070','RDW','PDW','MPV','XH002309','XH002306','XH002310')            
   AND O.CLAS          = 4                                 
   AND R.RESULTFLAG    = 0                                 
 GROUP BY O.ACPTDATE,R.SPCMNO,P.PATID,P.PATNAME,P.SEX      
 ORDER BY R.SPCMNO
