SELECT R.RESULTITEMCODE as ITEM                      FROM registinfos O, resultofnum R                 WHERE O.acptdate = R.acptdate                        AND R.SPCMNO = '1907080019'                AND O.patid = R.patid                              AND O.acptseq = R.acptseq                          AND O.CLAS = 4                                     AND R.RESULTFLAG = 0                               AND R.RESULTITEMCODE in ('D1860','D1850','D2510050','D1870','CA','GGT','IP','D2263','D3022A','D3022','D2611','D2300','D2280','D2310','D1840','D1880','D1830','DB','D2613','LDL-C','D0113','RF')      ORDER BY R.RESULTITEMCODE
