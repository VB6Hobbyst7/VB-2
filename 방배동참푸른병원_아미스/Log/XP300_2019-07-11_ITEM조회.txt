SELECT R.RESULTITEMCODE as ITEM                      FROM registinfos O, resultofnum R                 WHERE O.acptdate = R.acptdate                        AND R.SPCMNO = '1907110002'                AND O.patid = R.patid                              AND O.acptseq = R.acptseq                          AND O.CLAS = 4                                     AND R.RESULTFLAG = 0                               AND R.RESULTITEMCODE in ('D0002010','D0002030','D0002050','D0002040','XH002005','XH002006','XH002007','D0002070','XH002008','D0002060','JCCP02','XH002309','XH002306','XH002310')      ORDER BY R.RESULTITEMCODE
