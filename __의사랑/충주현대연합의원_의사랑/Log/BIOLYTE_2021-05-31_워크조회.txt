Select DISTINCT        SAVESEQ                 
     , HOSPDATE                
     , INOUT                   
     , CHARTNO                 
     , BARCODE                 
     , PID                     
     , PNAME                   
     , PSEX                    
     , PAGE                    
     , PJUMIN                  
     , COUNT(EXAMCODE) AS CNT  
  From UB_PATRESULT                                            
 Where HOSPDATE Between '20210531' AND '20210531'      
   And EXAMCODE IN ('C3791','C3791000','C3792','C3792000','C3793','C3793000')                        
 Group By SAVESEQ,HOSPDATE,INOUT,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN 
 Order by HOSPDATE,PID,PNAME,BARCODE
