SELECT a.COMEXAMID, a.EXAMID, b.EXAMNAME, a.RESULT, a.EXAMNO, a.TRANSYN, b.reflow,b.refhigh,b.seqno 
  FROM PAT_RES a, EQUIPEXAM b 
 WHERE a.EXAMID = b.EXAMCODE 
   AND a.COMEXAMID = b.EQUIPCODE 
   AND a.BARCODE = '06122000186792' 
   AND a.EXAMTYPE = 'C' 
 GROUP BY a.EXAMNO, a.COMEXAMID, a.EXAMID, b.EXAMNAME, a.RESULT, a.TRANSYN, b.reflow,b.refhigh  ORDER BY b.seqno * 10
