Select DISTINCT        SAVESEQ                 
     , HOSPDATE                
     , INOUT                   
     , CHARTNO                 
     , BARCODE                 
     , PID                     
     , PNAME                   
     , PSEX                    
     , PAGE                    
     , PJUMIN                  
     , COUNT(EXAMCODE) AS CNT  
  From UB_PATRESULT                                            
 Where HOSPDATE Between '20000902' AND '20200902'      
   And EXAMCODE IN ('3001','3008','3011','3012','3013','3013-1','3014','3015','ANC101','ANC101-1','cbc52','cbc53','cbc54','cbc55','H1234','H1324-1','Hb','MCH','MCHC','MCV','PLT','RBC','RDW')                        
   And (RESULT = '' OR RESULT IS NULL) 
   AND SENDFLAG = '0'                  
 Group By SAVESEQ,HOSPDATE,INOUT,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN 
 Order by HOSPDATE,PID,PNAME,BARCODE
