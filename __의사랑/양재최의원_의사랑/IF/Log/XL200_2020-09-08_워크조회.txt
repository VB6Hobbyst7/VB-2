Select DISTINCT        SAVESEQ                 
     , HOSPDATE                
     , INOUT                   
     , CHARTNO                 
     , BARCODE                 
     , PID                     
     , PNAME                   
     , PSEX                    
     , PAGE                    
     , PJUMIN                  
     , COUNT(EXAMCODE) AS CNT  
  From UB_PATRESULT                                            
 Where HOSPDATE Between '20200908' AND '20200908'      
   And EXAMCODE IN ('ALB','ALP','BUN','CRE','GGT','GLU','GOT','GPT','HDL','LDL','TB','TC','TG','TP','UA')                        
   And (RESULT = '' OR RESULT IS NULL) 
   AND SENDFLAG = '0'                  
 Group By SAVESEQ,HOSPDATE,INOUT,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN 
 Order by HOSPDATE,PID,PNAME,BARCODE
