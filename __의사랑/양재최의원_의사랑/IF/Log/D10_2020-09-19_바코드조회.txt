Select DISTINCT        SAVESEQ                 
     , HOSPDATE                
     , INOUT                   
     , CHARTNO                 
     , BARCODE                 
     , PID                     
     , PNAME                   
     , PSEX                    
     , PAGE                    
     , PJUMIN                  
     , EXAMCODE        AS ITEM 
  From UB_PATRESULT                        
 Where BARCODE = '091910-0010368'      
   And PID     = 'o9601'    
   And EXAMCODE IN ('D3061')    
   And (RESULT = '' OR RESULT IS NULL) 
   AND SENDFLAG = '0'                  
 Order by SAVESEQ,HOSPDATE,PNAME
