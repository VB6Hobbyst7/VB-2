Select DISTINCT        SAVESEQ                      , HOSPDATE                     , INOUT                        , CHARTNO                      , BARCODE                      , PID                          , PNAME                        , PSEX                         , PAGE                         , PJUMIN                       , COUNT(EXAMCODE) AS CNT    From UB_PATRESULT                                             Where HOSPDATE Between '20190902' AND '20190910'         And EXAMCODE IN ('B1050','B1040','B1010','B1010-1','B1020','mcv','B1020B','B1020C','B1220','B1060','B1060B','B1230','pdw','lymph','mono','B1091','RBC-IN','B1091')                           And (RESULT = '' OR RESULT IS NULL)                          Group By SAVESEQ,HOSPDATE,INOUT,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN  Order by HOSPDATE,PID,PNAME,BARCODE
