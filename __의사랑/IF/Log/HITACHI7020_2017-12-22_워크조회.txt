Select Distinct SAVESEQ, HOSPDATE,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN,COUNT(EXAMCODE) AS CNT   From PATRESULT  Where HOSPDATE Between '20170122' AND '20171222'   and EXAMCODE IN ('l5-1','l5-3','l1','l2','l5','l6','l6-1','l6-3','l13','l13-1','l3','l7','l12','l16','l16-1','l16-3','l18','l18-1','l17','l17-1','g1','g2','l9','l9-1','l9-2','l9-3','l10','l11','l11-1')     Group By SAVESEQ,HOSPDATE,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN  Order by SAVESEQ,HOSPDATE,PNAME
