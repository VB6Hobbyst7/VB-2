Select Distinct SAVESEQ, HOSPDATE,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN,COUNT(EXAMCODE) AS CNT   From PATRESULT  Where HOSPDATE Between '20170402' AND '20180402'   and EXAMCODE IN ('C3791','C3792','C3793')       and (RESULT = '' OR RESULT IS NULL)  Group By SAVESEQ,HOSPDATE,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN  Order by SAVESEQ,HOSPDATE,PNAME
