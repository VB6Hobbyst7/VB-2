Select Distinct SAVESEQ, HOSPDATE,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN,COUNT(EXAMCODE) AS CNT   From PATRESULT  Where HOSPDATE Between '20180323' AND '20180323'   and EXAMCODE IN ('C3791','C3792','C3793')     Group By SAVESEQ,HOSPDATE,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN  Order by SAVESEQ,HOSPDATE,PNAME
