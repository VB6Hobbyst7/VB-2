Select DISTINCT        SAVESEQ                 
     , HOSPDATE                
     , INOUT                   
     , CHARTNO                 
     , BARCODE                 
     , PID                     
     , PNAME                   
     , PSEX                    
     , PAGE                    
     , PJUMIN                  
     , EXAMCODE        AS ITEM 
  From UB_PATRESULT                        
 Where BARCODE = '10100000046844'      
   And PID     = 'o42057'    
   And EXAMCODE IN ('b1010','b1020','b1040','b1050','b1060','lym','mch','mchc','mcv','mono','seg')    
   And (RESULT = '' OR RESULT IS NULL) 
   AND SENDFLAG = '0'                  
 Order by SAVESEQ,HOSPDATE,PNAME
