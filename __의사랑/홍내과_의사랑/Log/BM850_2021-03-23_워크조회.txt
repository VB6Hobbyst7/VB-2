Select DISTINCT        SAVESEQ                 
     , HOSPDATE                
     , INOUT                   
     , CHARTNO                 
     , BARCODE                 
     , PID                     
     , PNAME                   
     , PSEX                    
     , PAGE                    
     , PJUMIN                  
     , COUNT(EXAMCODE) AS CNT  
  From UB_PATRESULT                                            
 Where HOSPDATE Between '20210323' AND '20210323'      
   And EXAMCODE IN ('1300','1311','1312','2010-1','2010-2','2010-3','D0002010','D0002030','D0002040','D0002050','D0002070')                        
   And (RESULT = '' OR RESULT IS NULL)                     
   AND SENDFLAG = '0'                                      
 Group By SAVESEQ,HOSPDATE,INOUT,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN 
 Order by HOSPDATE,PID,PNAME,BARCODE
