 SELECT a.spcyy AS SPCYY, a.spcno AS SPCNO FROM s2lab201 a, s2lab302 b, his001_v c, s2lab032 d,s2lab001 e, s2com006 f, s2ord102 g WHERE a.rcvdt BETWEEN '20171128' AND '20171129' AND a.stscd IN ('2', '3')     AND a.workarea=b.workarea AND a.accdt=b.accdt AND a.accseq=b.accseq     AND b.testcd IN (SELECT TESTCD FROM s2lab703 WHERE eqpcd='E001')     AND (b.rstcd IS NULL OR b.rstcd='')     AND (b.vfydt IS NULL OR b.vfydt='')     AND b.ptid=g.ptid AND b.orddt=g.orddt AND b.ordno=g.ordno AND b.ordseq=g.ordseq     AND a.ptid=c.ptid AND d.cdindex='C215' AND a.spccd=d.cdval1     AND b.testcd=e.testcd     AND e.applydt=(SELECT MAX(applydt) FROM s2lab001 WHERE testcd=b.testcd)     AND a.rcvid=f.empid ORDER BY a.workarea, a.accdt, a.accseq
