select distinct m.workarea                  , m.accdt AS HOSPDATE                  , m.accseq                             , m.spcyy                              , m.spcno                              , m.ptid AS PID                        , p.수진자명 AS PNAME                  , m.rcvdt                              , m.rcvtm                              , COUNT(r.testcd) AS CNT            from plis..s2lab201 m                      , medichart..TB_인적사항 p                  , plis..s2lab302 r                      , plis..s2lab001 e                  where SUBSTRING(m.accdt,1,8) BETWEEN '20171201' AND '20171228'   and m.workarea = '01'     and r.testcd IN ('B1050','B1040','B1010','B1020','b1010a','b1010b','b1010c','B1220','B1060','B1060A','PCT','B1230','B1091C','B1091A','B1050-3','B1050-4','B1050-5')       and (r.vfydt IS NULL OR r.vfydt='')      and m.ptid = p.챠트번호                    and m.workarea = r.workarea                and m.accdt = r.accdt                      and m.accseq = r.accseq                    and r.testcd = e.testcd                   Group by m.workarea, m.accdt, m.spcyy,m.spcno,m.accseq, m.ptid,p.수진자명,m.rcvdt, m.rcvtm   Order by m.rcvdt, m.rcvtm
