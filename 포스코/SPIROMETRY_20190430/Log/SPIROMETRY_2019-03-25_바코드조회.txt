SELECT a.per_name          AS PNAME                                     , a.per_jikbun        AS BARCODE                                   , b.meditem           AS ITEM                                      , MAX(a.per_gumjin_date)   AS HOSPDATE                               FROM TB_JUPSU a                                                       , TB_JUPSU_ITEM b                                                  , BAG_CODEVALUE c                                              WHERE a.com_code = '1'                                               AND a.per_jikbun            = '629602'                 AND a.per_ssn               = b.per_ssn                            AND a.per_jupsu_date        = b.per_jupsu_date                     AND a.per_ssn               = b.per_ssn                            AND a.per_gumjin_date       = b.per_gumjin_date                    AND b.meditem               IN ( '06000','06003','06001','N/A','N/A','06004','N/A','06011','N/A','06006','N/A','99999','N/A','N/A','N/A','N/A','N/A','06012','N/A','N/A','N/A','N/A','N/A','N/A','N/A','N/A','N/A')               AND 'P00' || a.jupsu_gubun  = c.gumsacode                          AND c.codegubun = 'P00'                                          GROUP BY a.per_name,  a.per_jikbun, b.meditem
