SELECT DISTINCT       a.per_gumjin_date   AS HOSPDATE                                  , a.per_name          AS PNAME                                     , a.per_jikbun        AS BARCODE                                   , a.per_jupsu_date                                                 , a.per_jupsu_time                                                 , COUNT(b.meditem)    AS CNT                                    FROM TB_JUPSU a                                                    , TB_JUPSU_ITEM b                                               , BAG_CODEVALUE c                                           WHERE a.com_code = '2'                                            AND a.per_gumjin_date BETWEEN '20190222' AND '20190222'    AND a.per_jupsu_date        = b.per_jupsu_date                     AND a.per_ssn               = b.per_ssn                            AND a.per_gumjin_date       = b.per_gumjin_date                    AND b.meditem               IN ( '000006000','000006003','000006001','N/A','N/A','000006004','N/A','N/A','N/A','000006006','N/A','N/A','N/A','N/A','N/A','N/A','N/A','000006012','N/A','N/A','N/A','N/A','N/A','N/A','N/A','N/A','N/A')               AND 'P00' || a.jupsu_gubun  = c.gumsacode                          AND c.codegubun = 'P00'                                          GROUP BY a.per_gumjin_date, a.per_name, a.per_jikbun, a.per_jupsu_date, a.per_jupsu_time  ORDER BY a.per_jupsu_date, a.per_jupsu_time
