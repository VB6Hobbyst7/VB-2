 SELECT DISTINCT R.ResLabCod AS ITEM                   
   FROM RESINF AS R                                    
 WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '848140945'
   AND R.RESLABCOD IN ('B25701','B25801','B27101','C37111','C24111','C24431','C37501','C24201')             
   AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')     
   AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)
