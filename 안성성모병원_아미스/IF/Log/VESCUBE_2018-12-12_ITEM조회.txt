SELECT R.RESULTITEMCODE as ITEM                      FROM registinfos O, resultofnum R                 WHERE O.acptdate = R.acptdate                        AND R.SPCMNO = '1812120043'                AND O.patid = R.patid                              AND O.acptseq = R.acptseq                          AND O.CLAS = 4                                     AND R.RESULTFLAG = 0                               AND R.ORDERCODE IN ('LH0007','LH0019','LH0026')             AND R.RESULTITEMCODE in ('XH0061')      ORDER BY R.RESULTITEMCODE
