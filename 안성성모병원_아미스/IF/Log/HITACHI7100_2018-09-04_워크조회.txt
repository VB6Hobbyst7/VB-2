SELECT DISTINCT        a.RECEIPTDATE    AS HOSPDATE      , a.SPECIMENNUM    AS BARCODE       , a.RECEIPTNO      AS CHARTNO       , a.IPDOPD         AS INOUT         , a.PTNO           AS PID           , a.SNAME          AS PNAME         , COUNT(a.LABCODE) AS CNT         FROM SLA_LabMaster a, SLA_LabResult b    WHERE a.RECEIPTNO     = b.RECEIPTNO          AND a.ORDERCODE     = b.ORDERCODE          and a.RECEIPTDATE   = b.RECEIPTDATE        AND a.SPECIMENNUM   = b.SPECIMENNUM        AND a.RECEIPTDATE BETWEEN '2018-09-04' and '2018-09-04'   AND b.LABCODE IN ('C2200','C2210','C37111','B2570','B2580','B2602','C2411','C2420','C3730','C3750','C3720','B2710','C2443')      AND a.JSTATUS < '3'                      GROUP BY a.RECEIPTDATE, a.SPECIMENNUM, a.RECEIPTNO, a.IPDOPD, a.PTNO, a.SNAME  ORDER BY a.RECEIPTDATE,a.SPECIMENNUM
