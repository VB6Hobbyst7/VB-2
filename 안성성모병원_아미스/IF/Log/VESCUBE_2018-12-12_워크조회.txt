SELECT DISTINCT       O.ACPTDATE              as HOSPDATE      , R.SPCMNO                as BARCODE       , P.PATID                 as PID           , P.PATNAME               as PNAME         , P.SEX                   as SEX           , COUNT(R.RESULTITEMCODE) as CNT        FROM REGISTINFOS O, RESULTOFNUM R, PATMST P                   WHERE O.ACPTDATE  = R.ACPTDATE                                   AND O.PATID     = R.PATID                                      AND O.ACPTSEQ   = R.ACPTSEQ                                    AND O.PATID     = P.PATID                                      AND O.ACPTDATE BETWEEN '20181212' and '20181212'       AND R.RESULTITEMCODE IN ('XH0061')                   AND R.ORDERCODE      IN ('LH0007','LH0019','LH0026')                    AND O.CLAS          = 4                                        AND R.RESULTFLAG    = 0                                      GROUP BY O.ACPTDATE,R.SPCMNO,P.PATID,P.PATNAME,P.SEX           ORDER BY R.SPCMNO
