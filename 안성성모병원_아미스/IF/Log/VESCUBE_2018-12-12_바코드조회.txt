SELECT DISTINCT       O.ACPTDATE              as HOSPDATE      , R.SPCMNO                as BARCODE       , P.PATID                 as PID           , P.PATNAME               as PNAME         , P.SEX                   as SEX           , R.ORDERCODE             as ORDCODE       , R.RESULTITEMCODE        as ITEM       FROM REGISTINFOS O, RESULTOFNUM R, PATMST P       WHERE O.ACPTDATE  = R.ACPTDATE                       AND O.PATID     = R.PATID                          AND O.ACPTSEQ   = R.ACPTSEQ                        AND O.PATID     = P.PATID                          AND R.SPCMNO = '1812120043'                AND R.RESULTITEMCODE IN ('XH0061')       AND O.CLAS          = 4                            AND R.RESULTFLAG    = 0                          ORDER BY R.SPCMNO
