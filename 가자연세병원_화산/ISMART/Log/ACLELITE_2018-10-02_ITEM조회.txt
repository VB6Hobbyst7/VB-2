 SELECT DISTINCT R.ResLabCod AS ITEM                      FROM RESINF AS R                                     WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '검체번호'   AND R.RESLABCOD IN ('B1540','B153001','XXX','B1530','B1601','B1723')                AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')        AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)       Order By R.ResLabCod
