SELECT DISTINCT        O.ORDDT             as HOSPDATE      , O.SPCNO             as BARCODE       , O.PTID              as PID           , O.PTNM              as PNAME         , O.SEX               as SEX           , O.AGE               as AGE           , COUNT(T.TESTCD)     as CNT        FROM TC201 O, TC301 T                 WHERE O.SPCNO = T.SPCNO                  AND O.OrdDt between  '20181003' and '20181004'      And T.TESTCD in ('C3791','C3792','C3793') GROUP BY O.ORDDT,O.SPCNO,O.PTID,O.PTNM,O.SEX,O.AGE  Order By O.ORDDT,O.SPCNO
