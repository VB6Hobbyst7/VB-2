 SELECT DISTINCT         R.RESACPDTM                AS HOSPDATE                       , R.RESCHTNUM                AS CHARTNO                        , P.PBSPATNAM                AS PNAME                          , R.RESSPMNUM                AS BARCODE                        , COUNT(R.RESSHTNAM)         AS CNT                         FROM RESINF R, PBSINF P                                       WHERE R.RESACPDTE Between '20181001' AND '20181002'      AND R.RESCHTNUM    = P.PBSCHTNUM                               AND R.RESLABCOD IN ('B1540','B153001','XXX','B1530','B1601','B1723')                        AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')                AND (R.RESMZHMNT = ''  OR R.RESMZHMNT = ' '  OR R.RESMZHMNT IS NULL) GROUP BY R.RESACPDTM, R.RESCHTNUM, P.PBSPATNAM, R.RESSPMNUM    ORDER BY R.RESACPDTE,R.RESACPNUM, R.RESMZHTYP ASC, R.RESLABCOD ASC
