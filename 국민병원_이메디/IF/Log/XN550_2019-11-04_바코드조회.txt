Select DISTINCT        R.RESULTDATE        AS HOSPDATE 
     , P.PATID             AS PID      
     , P.PATNAME           AS PNAME    
     , R.SPCMNO            AS BARCODE  
     , R.RESULTITEMCODE    AS ITEM     
  From RESULTOFNUM R, REGISTINFOS L, PATMST P  
 Where R.SPCMNO        = 'QC-92191102'  
   And R.ACPTDATE      =   L.ACPTDATE          
   And R.ACPTSEQ       =   L.ACPTSEQ           
   And R.PATID         =   P.PATID             
   And R.RESULTITEMCODE IN ('XH002001','XH002002','XH002003','XH002004','XH002009','XH002005','XH002006','XH002007','XH002008','XH002306','XH002309','XH002310','XH002307','XH002308','XH002010','XH002012')
   And (R.TEXTRESULTVAL = '' OR R.TEXTRESULTVAL IS NULL)   
 ORDER BY R.RESULTDATE,P.PATID
