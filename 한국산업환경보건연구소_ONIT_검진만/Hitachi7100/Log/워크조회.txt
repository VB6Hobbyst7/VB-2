
SELECT DISTINCT PER_GUMJIN_DATE as 접수일자, PER_NAME as 이름, CHARTNO as 챠트번호, PER_GUM_NUM as 바코드번호, SEQNO, MEDITEM, INTERFACECODE as ITEM 
  FROM ONIT..GUMJIN_INTERFACE 
 WHERE PER_GUMJIN_DATE BETWEEN '20160725' AND '20160725' 
   AND INTERFACECODE IN ('C2200','1001','1002','C2210','C3720','1004','C3721','1003','1005','B2570','1006','B2580','1008','B2710','1007','B2602','C3780','1018','1009','C2411','1012','C2443','B2611','1010','1022','C2420','1013','C3711','1016','C3730','C3750','1017') 
   AND STATUS = '0' 
   AND RESULT = ''  OR RESULT IS NULL 
 ORDER BY 접수일자,챠트번호
