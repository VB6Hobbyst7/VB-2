
SELECT PER_GUMJIN_DATE as 접수일자, PER_NAME as 이름, PER_GUM_NUM as 바코드번호, EDPSCODE as ITEM 
  FROM ONIT..GUMJIN_INTERFACE 
 WHERE PER_GUMJIN_DATE BETWEEN '20210224' AND '20210224' 
   AND EDPSCODE IN ('563','564','565','566','1319','985','986','426','839','924','1409','571','960','570','998','999') 
   AND RESULT = ''  OR RESULT IS NULL 
 ORDER BY 접수일자, PER_GUM_NUM * 10
