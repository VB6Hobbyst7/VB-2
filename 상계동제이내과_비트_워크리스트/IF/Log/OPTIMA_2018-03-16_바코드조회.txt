SELECT R.SPCMNO as BARCODE, P.PATID as PID, P.PATNAME as PNAME, P.SEX as SEX, O.ACPTDATE as HOSPDATE , O.ACPTSEQ, O.RSVACPTSTATE, O.RESULTSTATE, O.DEPTCODE, O.ORDERDATE, O.SLIPNO, O.IOFLAG, O.ORDERCODE, O.ORDERNAME, R.RESULTFLAG, R.RESULTNO, R.RESULTITEMCODE as ITEM   FROM registinfos O, resultofnum R, PATMST P  WHERE O.acptdate = R.acptdate    AND R.SPCMNO = 'Clear'   AND R.resultitemcode in ('B0030-10','B0030-09','B0030-06','B0030-07','B0030-02','B0030-08','B0030-05','B0030-04','B0030-03','LEU')   AND O.patid = R.patid    AND O.acptseq = R.acptseq    AND O.patid = P.patid    AND O.CLAS = 4    AND R.ORDERCODE IN ('B2640G')    AND R.RESULTFLAG = 0   ORDER BY R.SPCMNO
