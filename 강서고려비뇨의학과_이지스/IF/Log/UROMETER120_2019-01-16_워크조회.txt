SELECT DISTINCT        a.ACC_YMD       AS HOSPDATE          , a.RECEPT_NO     AS BARCODE           , a.PTNT_NO       AS PID               , c.PTNT_NM       AS PNAME             , c.BIRTH_YMD     AS AGE               , c.SEX           AS SEX               , COUNT(a.ORD_CD) AS CNT            FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c             WHERE a.ACC_YMD between '20190116' AND '20190116'    AND a.ORD_CD IN ('B0030-10','B0030-09','B0030-06','B0030-07','B0030-02','B0030-08','B0030-05','B0030-04','B0030-03','LEU')                       AND a.STS_CD     = 'A'                                     AND a.SUTAK_CD   = ''                                      AND a.RECEPT_NO  = b.RECEPT_NO                             AND a.PTNT_NO    = c.PTNT_NO                             GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, c.PTNT_NM,c.BIRTH_YMD,c.SEX  ORDER BY a.ACC_YMD, a.PTNT_NO
