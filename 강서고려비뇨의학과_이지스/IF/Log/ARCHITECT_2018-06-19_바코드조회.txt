SELECT DISTINCT        a.RECEIPTDATE    AS HOSPDATE      , a.SPECIMENNUM    AS BARCODE       , a.RECEIPTNO      AS CHARTNO       , a.IPDOPD         AS INOUT         , a.PTNO           AS PID           , a.SNAME          AS PNAME         , a.ORDERCODE      AS ORDCODE       , b.LABCODE        AS ITEM        FROM SLA_LabMaster a, SLA_LabResult b    WHERE a.RECEIPTNO     = b.RECEIPTNO          AND a.ORDERCODE     = b.ORDERCODE          and a.RECEIPTDATE   = b.RECEIPTDATE        AND a.SPECIMENNUM   = b.SPECIMENNUM        AND a.SPECIMENNUM   = '02116050'   AND b.LABCODE IN ('C468242','C468246','C472361','C1234')      AND a.JSTATUS < '3'                      ORDER BY b.LABCODE
