SELECT DISTINCT        a.ACC_YMD       AS HOSPDATE          , a.RECEPT_NO     AS BARCODE           , a.PTNT_NO       AS PID               , c.PTNT_NM       AS PNAME             , c.BIRTH_YMD     AS AGE               , c.SEX           AS SEX               , COUNT(a.ORD_CD) AS CNT            FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c             WHERE a.ACC_YMD between '20201118' AND '20201118'    AND a.ORD_CD IN ('UA(비)-5','UA0-5','UA-5','UA(비)-2','UA0-2','UA-2','UA(비)-8','UA0-8','UA-8','UA(비)-3','UA0-3','UA-3','UA(비)-7','UA0-7','UA-7','UA(비)-9','UA0-9','UA-9','UA(비)-1','UA0-1','UA-1','UA(비)-6','UA0-6','UA-6','UA(비)-4','UA0-4','UA-4','UA(비)-0','UA-0','UA0-0')                       AND a.STS_CD     = 'A'                                     AND a.SUTAK_CD   = ''                                      AND a.RECEPT_NO  = b.RECEPT_NO                             AND a.PTNT_NO    = c.PTNT_NO                             GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, c.PTNT_NM,c.BIRTH_YMD,c.SEX  ORDER BY a.ACC_YMD, a.PTNT_NO
