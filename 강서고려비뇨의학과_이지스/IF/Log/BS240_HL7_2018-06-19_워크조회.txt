Select DISTINCT        SAVESEQ                      , HOSPDATE                     , INOUT                        , CHARTNO                      , BARCODE                      , PID                          , PNAME                        , PSEX                         , PAGE                         , PJUMIN                       , COUNT(EXAMCODE) AS CNT    From UB_PATRESULT                                             Where HOSPDATE Between '2018-01-04' AND '2018-06-19'         And EXAMCODE IN ('C2243','C2200','C2210','B2602','C3720','B2570','B2580','C2411','C2420','C2443','C3711','C3730','C3750','B2710','C3780','C2430A')                           And (RESULT = '' OR RESULT IS NULL)                          Group By SAVESEQ,HOSPDATE,INOUT,CHARTNO,BARCODE,PID,PNAME,PSEX,PAGE,PJUMIN  Order by SAVESEQ,HOSPDATE,PNAME
