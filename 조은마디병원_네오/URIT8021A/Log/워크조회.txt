 SELECT DISTINCT a.MEDM_ID,a.CHAM_INDEX,a.WORK_DATE, '입원' as IO,b.CHAM_NAME,b.CHAM_SEX,b.CHAM_YY 
  From E_ORDER..ORDER_IN2017 a, E_BASECODE..HP_CHAM b 
 Where a.WORK_DATE between '20170430' AND '20170502'   And a.CHAM_INDEX = b.CHAM_INDEX    AND a.CODE IN ('C3825')   AND a.TRANS = '2'  UNION ALL  SELECT DISTINCT a.MEDM_ID,a.CHAM_INDEX,a.WORK_DATE, '외래' as IO,b.CHAM_NAME,b.CHAM_SEX,b.CHAM_YY 
  From E_ORDER..ORDER_OUT2017 a, E_BASECODE..HP_CHAM b 
 Where a.WORK_DATE between '20170430' AND '20170502'   And a.CHAM_INDEX = b.CHAM_INDEX    AND a.CODE IN ('C3825')   AND a.TRANS = '2'  ORDER BY a.WORK_DATE, IO
