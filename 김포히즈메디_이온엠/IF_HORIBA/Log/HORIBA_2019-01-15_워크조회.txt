SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE       ,O.H141_TSAMPLENO     AS BARCODE        ,O.H141_SEQNO         AS PID            ,P.A110_CHARTNO       AS CHARTNO        ,P.A110_PATNM         AS PNAME          ,P.A110_JUMIN1        AS AGE            ,P.A110_SEX           AS SEX            ,COUNT(O.H141_SUGACD) AS CNT        FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P                 Where O.H141_ODRDAT BETWEEN '20190115' AND '20190115'    AND P.A110_ChartNo = O.H141_CHARTNO                            AND O.H141_NOTYYN  = 'N'                                       And O.H141_SUGACD IN ('D0002010','D0002030','D0002050','D0002040','B1060','LMCHC','B109103','B109102','B109104','D0002070','LSB109101','LSB109102','LSB109106','B1230','B109101','D0113')                    Group By O.H141_ODRDAT,O.H141_TSAMPLENO,O.H141_SEQNO,P.A110_CHARTNO,P.A110_PATNM,P.A110_JUMIN1,P.A110_SEX  Order By O.H141_ODRDAT, O.H141_SEQNO
