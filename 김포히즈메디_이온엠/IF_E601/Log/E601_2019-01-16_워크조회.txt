SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE       ,O.H141_TSAMPLENO     AS BARCODE        ,O.H141_SEQNO         AS PID            ,P.A110_CHARTNO       AS CHARTNO        ,P.A110_PATNM         AS PNAME          ,P.A110_JUMIN1        AS AGE            ,P.A110_SEX           AS SEX            ,COUNT(O.H141_SUGACD) AS CNT        FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P                 Where O.H141_ODRDAT BETWEEN '20190115' AND '20190116'    AND P.A110_ChartNo = O.H141_CHARTNO                            AND O.H141_NOTYYN  = 'N'                                       And O.H141_SUGACD IN ('D7002','D7018','D7026','AFP','D4311','CA-19-9','_0406','BAGE_C3340','D3230050','PP_C3340','D7001','D7015','D7212','_0396','D3230060','PSA','_0410','BAGE_C3360','D4902020')                    Group By O.H141_ODRDAT,O.H141_TSAMPLENO,O.H141_SEQNO,P.A110_CHARTNO,P.A110_PATNM,P.A110_JUMIN1,P.A110_SEX  Order By O.H141_ODRDAT, O.H141_SEQNO
