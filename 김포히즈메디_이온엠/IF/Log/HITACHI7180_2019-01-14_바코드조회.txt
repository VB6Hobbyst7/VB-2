SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE       ,O.H141_TSAMPLENO     AS BARCODE        ,P.A110_CHARTNO       AS CHARTNO        ,P.A110_PATNM         AS PNAME          ,P.A110_JUMIN1        AS AGE            ,P.A110_SEX           AS SEX            ,O.H141_SUGACD        AS ITEM           ,O.H141_SEQNO         AS SUBITEM    FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P   Where P.A110_ChartNo      = O.H141_CHARTNO         AND O.H141_TSAMPLENO    = '18000014311'   AND O.H141_NOTYYN       = 'N'                    And O.H141_SUGACD in ('B2570','B2580','B2710','C3711','C3750','C2411','C2443','C2420','C2430','C2200','C2210','B2602','C3720','C3721','C3730','C3780','B2611','B2630','C3795','C3794','B2590','C2490','C2510','C2241','C2243','C4901','CRF','C2500')      Order By O.H141_SUGACD
