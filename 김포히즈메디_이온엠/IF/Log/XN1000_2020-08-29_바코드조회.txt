SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE 
      ,O.H141_TSAMPLENO     AS BARCODE  
      ,P.A110_CHARTNO       AS CHARTNO  
      ,P.A110_PATNM         AS PNAME    
      ,P.A110_JUMIN1        AS AGE      
      ,P.A110_SEX           AS SEX      
      ,O.H141_SUGACD        AS ITEM     
      ,O.H141_SEQNO         AS SUBITEM  
  FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P, TB_H131_SPPRESULT R  
 Where P.A110_ChartNo      = O.H141_CHARTNO        
   AND O.H141_TSAMPLENO    = 'QC-01891101'  
   And O.H141_SUGACD in ('D0002010','NT-B1050','D0002030','NT-B1040','D0002050','NT-B1010','D0002040','NT-B1020','LSB102001','LSB102002','LSB102003','D0002070','NT-B1060','LSB109104','LSB109102','LSB109101','LSB109105','LSB109106','LSB106002','LSB102005','D0002020','LSB102004','D0002060','LSB102006')       
   And O.H141_SEQNO = R.H131_SEQNO                 
 Order By O.H141_SUGACD
