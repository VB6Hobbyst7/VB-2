SELECT DISTINCT O.H141_SUGACD AS ITEM                FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P     Where P.A110_ChartNo = O.H141_CHARTNO                AND O.H141_TSAMPLENO  = '2020003817'       AND O.H141_NOTYYN       IN ('N','T')                    And O.H141_SUGACD in ('D1003','LSB154002','LSB154001','LSB154003','D1004')        ORDER BY O.H141_SUGACD
