SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE       ,O.H141_TSAMPLENO     AS BARCODE        ,P.A110_CHARTNO       AS CHARTNO        ,P.A110_PATNM         AS PNAME          ,P.A110_JUMIN1        AS AGE            ,P.A110_SEX           AS SEX            ,O.H141_SUGACD        AS ITEM           ,O.H141_SEQNO         AS SUBITEM    FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P   Where P.A110_ChartNo      = O.H141_CHARTNO         AND O.H141_TSAMPLENO    = '4019000261'   AND O.H141_NOTYYN       = 'N'                    And O.H141_SUGACD in ('LSB003004','LSB003002','LSB003006','LSB003007','LSB003009','LSB003005','LSB003001','LSB003003','LSB003010','LSB003008')      Order By O.H141_SUGACD
