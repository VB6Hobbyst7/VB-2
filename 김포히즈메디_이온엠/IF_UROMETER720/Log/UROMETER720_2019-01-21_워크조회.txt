SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE       ,O.H141_TSAMPLENO     AS BARCODE        ,O.H141_SEQNO         AS PID            ,P.A110_CHARTNO       AS CHARTNO        ,P.A110_PATNM         AS PNAME          ,P.A110_JUMIN1        AS AGE            ,P.A110_SEX           AS SEX            ,COUNT(O.H141_SUGACD) AS CNT        FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P                 Where O.H141_ODRDAT BETWEEN '20190121' AND '20190121'    AND P.A110_ChartNo = O.H141_CHARTNO                            AND O.H141_NOTYYN  = 'N'                                       And O.H141_SUGACD IN ('LSB003004','LSB003002','LSB003006','LSB003007','LSB003009','LSB003005','LSB003001','LSB003003','LSB003010','LSB003008')                    Group By O.H141_ODRDAT,O.H141_TSAMPLENO,O.H141_SEQNO,P.A110_CHARTNO,P.A110_PATNM,P.A110_JUMIN1,P.A110_SEX  Order By O.H141_ODRDAT, O.H141_SEQNO
