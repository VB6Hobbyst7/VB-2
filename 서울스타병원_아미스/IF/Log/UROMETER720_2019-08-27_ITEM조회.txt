SELECT R.RESULTITEMCODE as ITEM                      FROM registinfos O, resultofnum R                 WHERE O.acptdate = R.acptdate                        AND R.SPCMNO = '1908270074'                AND O.patid = R.patid                              AND O.acptseq = R.acptseq                          AND O.CLAS = 4                                     AND (R.NUMRESULTVAL = '' OR R.NUMRESULTVAL IS NULL)     
   AND R.RESULTITEMCODE in ('XU000203','XU000204','XU000205','XU000211','XU000209','XU000206','XU000207','XU000210','XU000208','XU000212')      ORDER BY R.RESULTITEMCODE
