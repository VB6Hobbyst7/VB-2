SELECT R.RESULTITEMCODE as ITEM                      FROM registinfos O, resultofnum R                 WHERE O.acptdate = R.acptdate                        AND R.SPCMNO = '1907310004'                AND O.patid = R.patid                              AND O.acptseq = R.acptseq                          AND O.CLAS = 4                                     AND (R.NUMRESULTVAL = '' OR R.NUMRESULTVAL IS NULL)     
   AND R.RESULTITEMCODE in ('XC0027','XC0028','XC0029','XC0038','XC0030','LX0023','LCX16','LC727','XC0025','XC0047','XC0031','XC0032','XC0034','XC0021','XC00038','XC0048','XC100802','XC0022')      ORDER BY R.RESULTITEMCODE
