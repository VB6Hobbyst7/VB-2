   FROM RESINF AS R                                    
 WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '70006'
   AND R.RESLABCOD IN ('D2420023','D3230053','D3230063','D3250013','d4023013','D4040003','D4062013','D4290003','D4300033','D4311003','D4350003','D7011013','D7015003','D7018003')             
   AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')     
   AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)
