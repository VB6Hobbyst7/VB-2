SELECT DISTINCT        b.OBODORDT          as HOSPDATE      , b.OBODMRNO          as BARCODE       , b.OBODCASE          as CHARTNO       , b.OBODORNO          as PID           , b.OBODIOGB          as INOUT         , a.APATNAME          as PNAME         , a.APATPSEX          as SEX           , a.APATJMN1          as JUMIN1        , a.APATJMN2          as JUMIN2        , COUNT(b.OBODCODE)   as CNT        FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND b.OBODORDT between '20210113' AND '20210113'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('23AM|','23AN|','23AK|','23AP|','23AR|','23AD|','33BI1|','GLUCOS|','23BK|','23AE|','23BB|','23AF|','23AH|','23AI|','23AJ|001','23AG|','20AU|','23BI|','24AB|','23BC|','C2510|')    AND b.OBODSTAT = 'AC'  GROUP BY b.OBODORDT, b.OBODMRNO, b.OBODCASE, b.OBODORNO, b.OBODIOGB, a.APATNAME, a.APATPSEX, a.APATJMN1, a.APATJMN2 Order by b.OBODORDT, b.OBODMRNO
