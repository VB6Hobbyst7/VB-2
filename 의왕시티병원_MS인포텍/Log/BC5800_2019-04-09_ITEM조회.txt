Select DISTINCT ORCD AS ITEM       
  From emr.LRESULT                 
 Where PAID = '00196322'   
   And SPNO =  '201904090143'
   And ORCD IN ('B1050','B1040','B1040-1','B1010','B1020','B1060','B1060-1','LCBC-1','LCBC-2','LCBC-3','LCBC-4','B1091-2','B1091-1','B1091-4','B1091-5','B1091-3')
   And OKFL <> 'Y'                 
 Order By ORCD
