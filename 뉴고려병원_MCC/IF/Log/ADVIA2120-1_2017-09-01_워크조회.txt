SELECT DISTINCT  READING_YMD AS HOSPDATE, BCODE_NO AS BARCODE, PTNT_NO AS PID ,PTNT_NM AS PNAME ,AGE AS AGE ,SEX AS SEX ,IO_GB AS INOUT , COUNT(ORD_CD) AS CNT   FROM LIS_INTERFACE1_V  WHERE READING_YMD between '20170831' AND '20170901'   AND ORD_CD IN ('B1050','B1040','B1010','B1020','B1020N1','B1020N2','B1020N3','B1220','HDW','B1060','B1060N2','B1230','B1060N1','B1091N3','B1091N1','B1091N2','B1091N4','B1091N5','LUC','B1081','NRBC','B1050N3','B1050N1','B1050N2','B1050N4','B1050N5','B1050N7','NRBC#','LI#','MPXI#','WBCP#','CHCM','LARGE PLT')    AND STS_CD = '0' GROUP BY READING_YMD,BCODE_NO,PTNT_NO,PTNT_NM,AGE,SEX,IO_GB  ORDER BY READING_YMD,PTNT_NO,BCODE_NO
