SELECT DISTINCT  READING_YMD AS HOSPDATE, BCODE_NO AS BARCODE, PTNT_NO AS PID ,PTNT_NM AS PNAME ,AGE AS AGE ,SEX AS SEX ,IO_GB AS INOUT , COUNT(ORD_CD) AS CNT   FROM LIS_INTERFACE1_V  WHERE READING_YMD between '20170904' AND '20170904'   AND ORD_CD IN ('C3815N1','C3815N2','C3815N3','C3815N4','C3815N5','C3815N6','C3815N7','C3815N8','C3815N9','C3796N1')  GROUP BY READING_YMD,BCODE_NO,PTNT_NO,PTNT_NM,AGE,SEX,IO_GB  ORDER BY READING_YMD,PTNT_NO,BCODE_NO
