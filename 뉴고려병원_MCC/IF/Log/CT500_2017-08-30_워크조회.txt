SELECT DISTINCT  READING_YMD AS HOSPDATE, BCODE_NO AS BARCODE, PTNT_NO AS PID ,PTNT_NM AS PNAME ,AGE AS AGE ,SEX AS SEX ,IO_GB AS INOUT , COUNT(ORD_CD) AS CNT   FROM LIS_INTERFACE1_V  WHERE READING_YMD between '20170830' AND '20170830'   AND ORD_CD IN ('UA03','UA05','UA06','UA07','UA04','UA08','UA02','UA01','UA09','UA17','UA15','UA10')    AND STS_CD = '0' GROUP BY READING_YMD,BCODE_NO,PTNT_NO,PTNT_NM,AGE,SEX,IO_GB  ORDER BY READING_YMD,PTNT_NO,BCODE_NO
