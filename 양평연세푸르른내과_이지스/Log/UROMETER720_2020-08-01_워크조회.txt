SELECT DISTINCT        a.ACC_YMD       AS HOSPDATE     
     , a.RECEPT_NO     AS BARCODE      
     , a.PTNT_NO       AS PID          
     , c.PTNT_NM       AS PNAME        
     , c.BIRTH_YMD     AS AGE          
     , c.SEX           AS SEX          
     , COUNT(a.ORD_CD) AS CNT          
  FROM H3LAB_RESULT a                  
     , H1OPDIN b                       
     , HZ_MST_PTNT c                   
 WHERE a.ACC_YMD between '20200801' AND '20200801' 
   AND a.ORD_CD IN ('259-1','259-10','259-2','259-3','259-4','259-5','259-6','259-7','259-8','259-9','800-1','800-2','800-3','800-4','800-5','800-6','800-7','URINE4')                    
   AND a.STS_CD     = 'A'                                  
   AND a.RECEPT_NO  = b.RECEPT_NO                          
   AND a.PTNT_NO    = c.PTNT_NO                            
 GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, c.PTNT_NM,c.BIRTH_YMD,c.SEX 
 ORDER BY a.ACC_YMD, a.PTNT_NO
