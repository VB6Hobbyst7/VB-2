SELECT P.PATID as 챠트번호, P.PATNAME as 이름, P.SEX, O.ACPTDATE as 접수일자, O.ACPTSEQ, O.RSVACPTSTATE, O.RESULTSTATE, O.DEPTCODE, O.ORDERDATE, O.SLIPNO, O.IOFLAG, O.ORDERCODE, O.ORDERNAME, R.SPCMNO as 바코드번호, R.RESULTFLAG, R.RESULTNO, R.RESULTITEMCODE as ITEM   FROM registinfos O, resultofnum R, PATMST P  WHERE O.acptdate = R.acptdate    AND O.acptdate between '20170818' and '20170818'   AND R.resultitemcode in ('BE(B)','Beecf','Ca++','HCO3-','HCO3std','Hct','K+','Na+','SO2c','TCO2','THbc','pCO2','pH','pO2','Temp')   AND O.patid = R.patid    AND O.acptseq = R.acptseq    AND O.patid = P.patid    AND O.CLAS = 4    AND O.ORDERCODE IN ('C3815','C3815-1')   AND R.RESULTFLAG = 0   ORDER BY R.SPCMNO
