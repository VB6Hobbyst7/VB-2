   FROM RESINF AS R                                    
 WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '376569'
   AND R.RESLABCOD IN ('WBC','D0002013','b1050','RBC','D0002033','b1040','HB','D0002053','B1010','HCT','D0002043','B1020','MCV','B1020A','MCH','B1020B','MCHC','B1020C','RDW','PLT','D0002073','b1060','b1091','B1091B','B1091A','B1091C')             
   AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')     
   AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)
