SELECT DISTINCT        a.ACC_YMD   AS HOSPDATE 
     , a.RECEPT_NO AS BARCODE  
     , a.PTNT_NO   AS PID      
     , c.PTNT_NM   AS PNAME    
     , c.BIRTH_YMD AS AGE      
     , c.SEX       AS SEX      
     , a.ORD_CD    AS ITEM     
  FROM H3LAB_RESULT a          
     , H1OPDIN b               
     , HZ_MST_PTNT c           
 WHERE a.RECEPT_NO = '83791278511'  
   AND a.ORD_CD IN ('ANC','BIL','BLD','GLU','KET','LEU','NIT','pH','PRO','S.G','URO')
   AND a.RECEPT_NO = b.RECEPT_NO           
   AND a.PTNT_NO    = c.PTNT_NO            
 ORDER BY a.ORD_CD
