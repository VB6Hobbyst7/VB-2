SELECT DISTINCT        CONVERT(NVARCHAR(50),M.접수일자,112)    AS HOSPDATE 
     , M.접수번호                              AS PID      
     , M.차트번호                              AS CHARTNO  
     , M.성명                                  AS PNAME    
     , M.성별                                  AS SEX      
     , M.나이                                  AS AGE      
     , E.검사코드                              AS ITEM     
  FROM VW_검사접수 M, VW_검사결과 R, VW_검사코드 E  
 WHERE M.접수일자 = CONVERT(DATE, '20190703')
   AND E.학부코드 = 'U2' 
   AND E.검사코드 IN ('5379','5380','5381','5382','5383','5391','5385','5384','5386','5387','5388','5389','5390','5393','5394','5392','5395','5396','5397') 
   AND isnull(R.보고여부, 'N') <> 'Y' 
   AND (R.결과값 is null or R.결과값 = '') 
   AND M.접수일자 = R.접수일자 
   AND M.접수번호 = R.접수번호
