SELECT DISTINCT        R.pro_code          AS ITEM       FROM trust P, trures R  WHERE P.request_date  = '20190404'   AND R.exam_no       = '1146'
   AND R.pro_code      IN ('5379','1595','5617','1229','5380','5381','5382','5383','5391','5385','5384','5386','5387','5388','5389','5390','5393','5394','5392','5395','5396')    AND R.exam_code     <> 'X999'    AND P.request_date  = R.request_date    AND P.exam_no       = R.exam_no  ORDER BY R.pro_code
