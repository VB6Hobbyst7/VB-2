SELECT DISTINCT P.request_date AS 접수일자, P.exam_no AS 내원번호, P.company_code AS 의뢰처, P.chart_no AS 차트번호, p.personal_id, p.person_name AS 이름,        P.worker_code, P.patient_kind, P.person_sex AS 성별, P.person_age AS 나이,        R.exam_order, R.exam_code AS ITEM, E.exam_ename, R.pro_code AS 처방코드   FROM trust P, trures R, examitem E  WHERE P.request_date = '20170725'   AND P.exam_no = '7131'   AND R.pro_code IN ('5150','5152','5182')    AND R.exam_code <> 'X999'    AND P.request_date = R.request_date    AND P.exam_no = R.exam_no    AND R.exam_code = E.exam_code  ORDER BY P.request_date, P.exam_no
