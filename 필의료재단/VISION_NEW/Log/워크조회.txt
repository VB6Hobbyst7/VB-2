SELECT DISTINCT CONVERT(NVARCHAR(50),M.접수일자,112) AS 접수일자, M.접수번호 AS 내원번호,'' AS 의뢰처, M.차트번호 AS 차트번호, M.성명 AS 이름,       M.성별 AS 성별,M.나이 AS 나이, E.검사코드 AS 처방코드   FROM VW_검사접수 M, VW_검사결과 R, VW_검사코드 E   WHERE M.접수일자 = CONVERT(DATE, '20190712')   AND E.학부코드 = 'H2'    AND E.검사코드 IN ('3022')    AND isnull(R.보고여부, 'N') <> 'Y'    AND (R.결과값 is null or R.결과값 = '')    AND M.접수일자 = R.접수일자    AND M.접수번호 = R.접수번호  ORDER BY 접수일자, M.접수번호
