SELECT EQUIPCODE,EXAMCODE,RESULT,EQUIPRESULT,REFFLAG,PANICVALUE,DELTAVALUE,PSEX,SEQNO,PAGE,PID,DISKNO,POSNO,EXAMSUBCODE 
  FROM PATRESULT 
 WHERE EQUIPNO = 'Allergen'
   AND DISKNO  = 'FOOD'
   AND MID(EXAMDATE,1,8) = '20170427'
   AND BARCODE = '2017041400119' 
   AND SAVESEQ = 1 ORDER BY SEQNO * 10
