SELECT DISTINCT  a.ACC_YMD     AS HOSPDATE, a.RECEPT_NO  AS BARCODE, a.PTNT_NO    AS PID ,a.PTNT_NM    AS PNAME, COUNT(a.ORD_CD) AS CNT   FROM H3LAB_RESULT a, H1OPTIN b, HZ_MST_PTNT c  WHERE a.ACC_YMD between '20170915' AND '20170915'   AND a.ORD_CD IN ('LYB1050','LYB1040','LYB1010','LYB1020','MCV','MCH','MCHC','LYB1060','B2602','C2420','C3750','C2411','LYB10911','LYB10913','LYB10913')    AND a.STS_CD    = 'A'   AND a.SUTAK_CD  = ''   AND a.RECEPT_NO = b.RECEPT_NO    AND a.STS_CD    = b.STS_CD  GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, a.PTNT_NM  ORDER BY a.ACC_YMD, a.PTNT_NO
