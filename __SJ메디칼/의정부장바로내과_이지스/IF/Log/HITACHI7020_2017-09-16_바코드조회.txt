SELECT DISTINCT  a.ACC_YMD     AS HOSPDATE, a.RECEPT_NO  AS BARCODE, a.PTNT_NO    AS PID, c.PTNT_NM    AS PNAME, a.ORD_CD AS ITEM  ,a.SPC_CD AS SPCCD   FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c  WHERE a.RECEPT_NO = '36'   AND a.ORD_CD IN ('C2200','C2210','C3720','C2420','B2602','B2570','B2580','B2710','C3711','C3730','C3750','B2611','B2590','C2411','C2443','B2640','C2243')    AND a.STS_CD    = 'A'   AND a.SUTAK_CD  = ''   AND a.RECEPT_NO = b.RECEPT_NO    AND a.ptnt_no    = c.ptnt_no  ORDER BY a.ORD_CD
