SELECT DISTINCT  a.ACC_YMD     AS HOSPDATE, a.RECEPT_NO  AS BARCODE, a.PTNT_NO    AS PID, a.PTNT_NM    AS PNAME, a.ORD_CD AS ITEM  ,a.SPC_CD AS SPCCD   FROM H3LAB_RESULT a, H1OPTIN b, HZ_MST_PTNT c    AND a.RECEPT_NO = '20'   AND a.ORD_CD IN ('LYB1050','LYB1040','LYB1010','LYB1020','MCV','MCH','MCHC','B2602','LYB1060','C3750','C2411','C2420','LYB10911','LYB10913')    AND a.STS_CD    = 'A'   AND a.SUTAK_CD  = ''   AND a.RECEPT_NO = b.RECEPT_NO    AND a.STS_CD    = b.STS_CD  ORDER BY a.ORD_CD
