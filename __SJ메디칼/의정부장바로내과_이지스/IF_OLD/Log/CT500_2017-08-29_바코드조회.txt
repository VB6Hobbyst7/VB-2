SELECT DISTINCT  READING_YMD AS HOSPDATE, BCODE_NO AS BARCODE, PTNT_NO AS PID ,PTNT_NM AS PNAME ,AGE AS AGE ,SEX AS SEX ,IO_GB AS INOUT ,ORD_CD AS ITEM ,SP_CD AS SPCCD  FROM LIS_INTERFACE1_V  WHERE BCODE_NO = '2'   AND ORD_CD IN ('UA03','UA05','UA06','UA07','UA04','UA08','UA02','UA01','UA09','UA17','UA15','UA10')    AND STS_CD = '0' ORDER BY ORD_CD
