SELECT DISTINCT   L.LABSERIAL  ,L.LABATTEND as PID, L.LABCHTNUM as CHARTNO, L.LABODRDTE as HOSPDATE , M.MANADMFOR as IO
 ,M.MANRESNUM as JUMIN, M.MANPATNAM as PNAME , L.LABBARCOD as BARCODE, L.LABODRSTP as SEQ 
, COUNT(L.LABODRCOD) as CNT  FROM ME_LABDAT L, ME_DAT D, ME_MAN M
 WHERE L.LABCHTNUM  = ''   AND L.LABODRDTE = ''   AND L.LABKEYNUM = D.DATKEYNUM 
   AND L.LABATTEND = D.DATATTEND 
   AND L.LABATTEND = M.MANATTEND 
   AND L.LABCHTNUM = D.DATCHTNUM 
   AND L.LABCHTNUM = M.MANCHTNUM 
   AND L.LABODRDTE = D.DATODRDTE 
   AND L.LABODRCOD IN ('C2200','C2210','B2570','B2580','B2602','C3720','B2710','C2443','C3730','C3750','C2411','C3711','C2420','C----','C3780','C2243')
   AND (L.LABCANCEL = '' OR L.LABCANCEL IS NULL) 
   AND (L.LABRESULT = ''  OR L.LABRESULT IS NULL)
   AND L.LABENDDEP < '3' 
 GROUP BY L.LABSERIAL,L.LABATTEND,L.LABCHTNUM,L.LABODRDTE,M.MANADMFOR,M.MANRESNUM,M.MANPATNAM,L.LABBARCOD,L.LABODRSTP  ORDER BY L.LABODRDTE, L.LABATTEND, L.LABBARCOD Desc, L.LABINSNUM, L.LABODRCOD
