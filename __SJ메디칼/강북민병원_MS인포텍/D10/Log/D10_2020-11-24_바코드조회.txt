Select DISTINCT        a.ORDT          AS HOSPDATE 
     , a.SPNO          AS BARCODE  
     , a.PAID          AS PID      
     , a.NWNO          AS CHARTNO  
     , b.PANM          AS PNAME    
     , b.SEXS          AS SEX      
     , b.AGES          AS AGE      
     , a.ORCD          AS ITEM     
     , a.ORQN          AS SEQ      
  From LRESULT a, APATINF b            
 Where a.SPNO = '202011240429'   
   And a.PAID = '00056747' 
   And a.PAID = b.PAID                 
   And a.ORCD IN ('C3825A','CL-IN','K-IN','NA-IN')  
   And a.OKFL <> 'Y'                   
   AND (a.RSFL IS NULL OR a.RSFL = 'N' OR a.RSFL = '')
 Order By a.ORCD
