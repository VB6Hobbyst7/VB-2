SELECT R.ORDT          AS HOSPDATE                     
     , R.SPNO          AS BARCODE                      
     , R.PAID          AS PID                          
     , R.NWNO          AS CHARTNO                      
     , P.PANM          AS PNAME                        
     , P.SEXS          AS SEX                          
     , P.AGES          AS AGE                          
     , COUNT(R.ORCD)   AS CNT                          
  FROM emr.LRESULT R                                   
     , emr.APATINF P                                   
 WHERE R.ORDT BETWEEN '20201118' AND '20201118'
   AND R.PAID  = P.PAID                                
   AND R.OKFL  <> 'Y'                                  
   AND R.ORCD  IN ('CL-IN','HbA1c','K-IN','NA-IN')                 
   AND (R.RSFL IS NULL OR R.RSFL = 'N' OR R.RSFL = '') 
 GROUP BY R.ORDT,R.SPNO,R.PAID,R.NWNO,P.PANM,P.SEXS,P.AGES 
 ORDER BY R.ORDT, R.PAID, P.PANM
