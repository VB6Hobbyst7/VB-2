SELECT DISTINCT        a.ACC_YMD       AS HOSPDATE          , a.RECEPT_NO     AS BARCODE           , a.PTNT_NO       AS PID               , c.PTNT_NM       AS PNAME             , c.BIRTH_YMD     AS AGE               , c.SEX           AS SEX               , COUNT(a.ORD_CD) AS CNT            FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c             WHERE a.ACC_YMD between '20191018' AND '20191018'    AND a.ORD_CD IN ('B2570','B2580','B2710','C2411','C2443','C2420','C2430','C3730','C3750','C3711','C2200','C2210','B2602','C3720','B2590','C3721','B2611','C3780','B2621')                       AND a.STS_CD     = 'A'                                     AND a.SUTAK_CD   = ''                                      AND a.RECEPT_NO  = b.RECEPT_NO                             AND a.PTNT_NO    = c.PTNT_NO                             GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, c.PTNT_NM,c.BIRTH_YMD,c.SEX  ORDER BY a.ACC_YMD, a.PTNT_NO
