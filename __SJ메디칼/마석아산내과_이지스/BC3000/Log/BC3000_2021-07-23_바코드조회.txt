SELECT DISTINCT        a.ACC_YMD   AS HOSPDATE      , a.RECEPT_NO AS BARCODE       , a.PTNT_NO   AS PID           , c.PTNT_NM   AS PNAME         , c.BIRTH_YMD AS AGE           , c.SEX       AS SEX           , a.ORD_CD    AS ITEM       FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c  WHERE a.RECEPT_NO = '4'     AND a.ORD_CD IN ('N11001','N11004','N11005','N11006','N11007','N11008','N11009','N11012','N1100203','N1100204','N1100201')       AND a.STS_CD    = 'A'                      AND a.SUTAK_CD  = ''                       AND a.RECEPT_NO = b.RECEPT_NO              AND a.PTNT_NO    = c.PTNT_NO             ORDER BY a.ORD_CD
