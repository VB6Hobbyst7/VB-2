SELECT DISTINCT        RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) AS ITEM  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND b.OBODMRNO = '00000002'             AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('23AM|','23AN|','23AK|','23AP|','23AR|','23AD|','33BI1|','GLUCOS|','23BK|','23AE|','23BB|','23AF|','23AH|','23AI|','23AJ|001','23AG|','20AU|','23BI|','24AB|','23BC|','C2510|')    AND b.OBODSTAT = 'AC'  Order by ITEM
