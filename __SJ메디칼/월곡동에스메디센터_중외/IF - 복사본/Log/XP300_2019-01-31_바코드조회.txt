SELECT DISTINCT        a.RECEIPTDATE    AS HOSPDATE      , a.SPECIMENNUM    AS BARCODE       , a.RECEIPTNO      AS CHARTNO       , a.IPDOPD         AS INOUT         , a.PTNO           AS PID           , a.SNAME          AS PNAME         , a.ORDERCODE      AS ORDCODE       , b.LABCODE        AS ITEM        FROM SLA_LabMaster a, SLA_LabResult b    WHERE a.RECEIPTNO     = b.RECEIPTNO          AND a.ORDERCODE     = b.ORDERCODE          and a.RECEIPTDATE   = b.RECEIPTDATE        AND a.SPECIMENNUM   = b.SPECIMENNUM        AND a.PTNO     = ''   AND b.LABCODE IN ('D0002010','D0002030','D0002050','D0002040','D0002070','MCV','MCH','MCHC','B1220','MPV','PCT','B1230','B1091N1','LY%','LYM%','B1091N2','MID%','B1091N3','SEG')      AND a.RECEIPTDATE = ''   AND a.JSTATUS < '3'                      ORDER BY b.LABCODE
