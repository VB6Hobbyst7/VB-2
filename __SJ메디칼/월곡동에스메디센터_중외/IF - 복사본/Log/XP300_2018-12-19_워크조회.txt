SELECT DISTINCT        a.RECEIPTDATE    AS HOSPDATE      , a.SPECIMENNUM    AS BARCODE       , a.RECEIPTNO      AS CHARTNO       , a.IPDOPD         AS INOUT         , a.PTNO           AS PID           , a.SNAME          AS PNAME         , COUNT(a.LABCODE) AS CNT         FROM SLA_LabMaster a, SLA_LabResult b    WHERE a.RECEIPTNO     = b.RECEIPTNO          AND a.ORDERCODE     = b.ORDERCODE          and a.RECEIPTDATE   = b.RECEIPTDATE        AND a.SPECIMENNUM   = b.SPECIMENNUM        AND a.RECEIPTDATE BETWEEN '2018-12-19' and '2018-12-19'   AND b.LABCODE IN ('B1050','B1040','B1010','B1020','B1060-1','B1060-2','B1060-3','B1060-4','B1091-2','B1091-3','B1091-1','B1220','B1220','B1330','LCBC-6','P-LCR','LCBC-7')      AND a.JSTATUS < '3'                      GROUP BY a.RECEIPTDATE, a.SPECIMENNUM, a.RECEIPTNO, a.IPDOPD, a.PTNO, a.SNAME  ORDER BY a.RECEIPTDATE,a.SPECIMENNUM
