SELECT DISTINCT LABCODE AS ITEM            
   FROM SLA_Labresult                      
 WHERE LABCODE IN ('D0002010','D0002030','D0002050','D0002040','MCV','MCH','MCHC','D0002070','LYMPH','MID','GRAN','D0002020','D0002020','PDW','D0002060','MPV','PCT')     
   AND RECEIPTDATE = '2019-12-12'  
   AND SPECIMENNUM = '115'  
 ORDER BY LABCODE
