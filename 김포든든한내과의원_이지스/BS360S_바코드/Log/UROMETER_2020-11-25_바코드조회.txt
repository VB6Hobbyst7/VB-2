SELECT DISTINCT       O.ACPTDATE              AS HOSPDATE             
     , R.SPCMNO                AS BARCODE              
     , P.PATID                 AS PID                  
     , P.PATNAME               AS PNAME                
     , P.SEX                   AS SEX                  
     , R.ORDERCODE             AS ORDCODE              
     , R.RESULTITEMCODE        AS ITEM                 
  FROM REGISTINFOS O                                   
     , RESULTOFNUM R                                   
     , PATMST      P                                   
 WHERE O.ACPTDATE      = R.ACPTDATE                    
   AND O.PATID         = R.PATID                       
   AND O.ACPTSEQ       = R.ACPTSEQ                     
   AND O.PATID         = P.PATID                       
   AND O.CLAS          = 4                             
   AND R.SPCMNO        = '11'          
   AND (R.NUMRESULTVAL = '' OR R.NUMRESULTVAL IS NULL) 
   AND R.RESULTITEMCODE IN ('B0030-01','B0030-02','B0030-03','B0030-04','B0030-05','B0030-06','B0030-07','B0030-08','B0030-09','B0030-10')
