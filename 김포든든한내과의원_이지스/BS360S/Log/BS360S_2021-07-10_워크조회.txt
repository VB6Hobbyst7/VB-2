SELECT DISTINCT        a.ACC_YMD       AS HOSPDATE     
     , a.RECEPT_NO     AS BARCODE      
     , a.PTNT_NO       AS PID          
     , c.PTNT_NM       AS PNAME        
     , c.BIRTH_YMD     AS AGE          
     , c.SEX           AS SEX          
     , COUNT(a.ORD_CD) AS CNT          
  FROM H3LAB_RESULT a                  
     , H1OPDIN b                       
     , HZ_MST_PTNT c                   
 WHERE a.ACC_YMD between '20210710' AND '20210710' 
   AND a.ORD_CD IN ('B2590','B2611','B2710','C2420','C2430','C2443','D0113','D1830','D1830','D1850','D1860','D1870','D1880','D2280','D2300','D2310','D2611','D3022','D3022=','D6911','D7811')                    
   AND a.STS_CD     = 'A'                                  
   AND (a.RESULT_NM = '' OR a.RESULT_NM IS NULL)            
   AND a.SUTAK_CD   = ''                                   
   AND a.RECEPT_NO  = b.RECEPT_NO                          
   AND a.PTNT_NO    = c.PTNT_NO                            
 GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, c.PTNT_NM,c.BIRTH_YMD,c.SEX 
 ORDER BY a.ACC_YMD, a.PTNT_NO
