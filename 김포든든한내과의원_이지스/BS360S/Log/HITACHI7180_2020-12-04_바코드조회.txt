SELECT DISTINCT       O.ACPTDATE              AS HOSPDATE             
     , R.SPCMNO                AS BARCODE              
     , P.PATID                 AS PID                  
     , P.PATNAME               AS PNAME                
     , P.SEX                   AS SEX                  
     , R.ORDERCODE             AS ORDCODE              
     , R.RESULTITEMCODE        AS ITEM                 
  FROM REGISTINFOS O                                   
     , RESULTOFNUM R                                   
     , PATMST      P                                   
 WHERE O.ACPTDATE      = R.ACPTDATE                    
   AND O.PATID         = R.PATID                       
   AND O.ACPTSEQ       = R.ACPTSEQ                     
   AND O.PATID         = P.PATID                       
   AND O.CLAS          = 4                             
   AND R.SPCMNO        = '2012040301'          
   AND (R.NUMRESULTVAL = '' OR R.NUMRESULTVAL IS NULL) 
   AND R.RESULTITEMCODE IN (' XC0034','CZ246','XC00038','XC0021','XC0022','XC0025','XC0027','XC0028','XC0029','XC0030','XC0031','XC0032','XC0038','XC0047J','XC0048','XI0011')
