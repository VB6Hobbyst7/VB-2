SELECT DISTINCT        a.ACC_YMD   AS HOSPDATE      , a.RECEPT_NO AS BARCODE       , a.PTNT_NO   AS PID           , c.PTNT_NM   AS PNAME         , c.BIRTH_YMD AS AGE           , c.SEX       AS SEX           , a.ORD_CD    AS ITEM       FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c  WHERE a.RECEPT_NO = '33445'     AND a.ORD_CD IN ('B0020-7','B0020-4','B0020-5','B0020-8','B0020-3','B0020-0','B0020-2','B0020-6','B0020-1','B0020-9')       AND a.STS_CD    = 'A'                      AND a.SUTAK_CD  = ''                       AND a.RECEPT_NO = b.RECEPT_NO              AND a.PTNT_NO    = c.PTNT_NO             ORDER BY a.ORD_CD
