SELECT DISTINCT        a.ACC_YMD       AS HOSPDATE          , a.RECEPT_NO     AS BARCODE           , a.PTNT_NO       AS PID               , c.PTNT_NM       AS PNAME             , c.BIRTH_YMD     AS AGE               , c.SEX           AS SEX               , COUNT(a.ORD_CD) AS CNT            FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c             WHERE a.ACC_YMD between '20181009' AND '20181010'    AND a.ORD_CD IN ('B1050','B1040','B1010','B1020','MCV','MCH','MCHC','B1060','B1091-7','B1091-8','B1091-17','B1091-14','B1091-15','B1091-16')                       AND a.STS_CD     = 'A'                                     AND a.SUTAK_CD   = ''                                      AND a.RECEPT_NO  = b.RECEPT_NO                             AND a.PTNT_NO    = c.PTNT_NO                             GROUP BY a.ACC_YMD, a.RECEPT_NO, a.PTNT_NO, c.PTNT_NM,c.BIRTH_YMD,c.SEX  ORDER BY a.ACC_YMD, a.PTNT_NO
