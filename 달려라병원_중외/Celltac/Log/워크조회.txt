SELECT DISTINCT a.RECEIPTDATE as 접수일자, a.SPECIMENNUM as 바코드번호, a.RECEIPTNO as 챠트번호, a.IPDOPD, a.PTNO as 내원번호, a.SNAME as 이름, a.LABCODE as ITEM, a.ORDERCODE
  FROM SLA_LabMaster a,SLA_LabResult b 
 WHERE a.RECEIPTDATE between '2016-11-23' and '2016-11-23'
   AND a.RECEIPTNO = b.RECEIPTNO 
   AND a.OrderCode = b.OrderCode 
   AND b.LABCODE IN ('3001','B1050','B1040','B1010','B1020','B1060-1','B1060-2','B1060-3','B1220','B1060-4','B1220','B1330','LCBC-6','B1091-1','MONO%','GR%','B1091-2') 

   AND a.JSTATUS < '3'

   AND (b.RESULT = '                                                  ' OR b.RESULT IS NULL) 
  ORDER BY a.RECEIPTDATE
