Select DISTINCT        R.SPECIMEN_SER                                                                  AS BARCODE       , (SELECT DISTINCT BUNHO  FROM CPL2010 WHERE SPECIMEN_SER = '20180162386') AS PID           , (SELECT DISTINCT SUNAME FROM CPL2010 WHERE SPECIMEN_SER = '20180162386') AS PNAME         , R.HANGMOG_CODE                                                                  AS ORDCD         , E.JANGBI_OUT_CODE                                                               AS ITEM       From CPL3020 R, EQP1010 E                                 WHERE R.SPECIMEN_SER          = '20180162386'         AND NVL(R.CONFIRM_YN, 'N')  = 'N'                          AND E.JANGBI_CODE           = '203'       AND R.HANGMOG_CODE          = E.HANGMOG_CODE               And JANGBI_OUT_CODE  IN ('PCT')               AND E.JANGBI_OUT_CODE IS Not Null                        ORDER BY E.JANGBI_OUT_CODE
