SELECT DISTINCT        SLABW_INOUT             AS INOUT    
     , SLABWS_DATE             AS HOSPDATE      , CHAM_KEY                AS BARCODE       , CHAM_WHANJA             AS PNAME         , CHAM_JUMIN1             AS JUMIN1        , CHAM_JUMIN2             AS JUMIN2        , SPECI_DATE              AS CHARTNO      , SPECI_SEQNO             AS PID        FROM BRWONMU..WCHAM A                                                       INNER JOIN OSLABW B     ON A.CHAM_KEY = B.SLABW_CHAM                   INNER JOIN OSLABWS C    ON B.SLABW_DATE = C.SLABWS_DATE                                        AND B.slabw_dept = C.slabws_dept                                       AND B.slabw_cnt = C.slabws_cnt                                         AND B.slabw_slab = C.slabws_slab                                       AND C.slabws_date >= '20210402'    
                               AND C.slabws_date <= '20210402'      
                               And CONCAT(RTRIM(LTRIM(C.SLABWS_MOMU)),'|',RTRIM(LTRIM(C.SLABWS_SCNT)))      IN ('D0002010|1','D0002010|2','D0002010|3','D0002020|1','D0002030|2','D0002040|4','D0002040|5','D0002040|6','D0002040|7','D0002050|13','D0002050|3','D0002050|9','D0002060|2','D0002070|8','D0013|10','D0013|11','D0013|12','D0013|13','D0013|14','D0013|5','D0013|6') 
                               And (C.slabws_result = '' OR C.slabws_result IS NULL)
       INNER JOIN OSLABS E     ON C.SLABWS_SCNT = E.SLABS_CNT                                         AND C.slabws_slab = E.slabs_key                                        AND E.slabs_use  = 1                           INNER JOIN Ospecislab F ON B.slabw_cnt = F.specis_cnt                                          AND B.slabw_date = F.specis_date                                       AND B.slabw_dept = F.specis_dept               INNER JOIN OSPECIMEN S  ON A.cham_key = S.SPECI_CHAM                                           AND F.specis_date = S.speci_date                                       AND F.specis_seqno = S.speci_seqno       ORDER BY SPECI_DATE, SPECI_SEQNO
