SELECT DISTINCT        O.H141_ODRDAT       AS HOSPDATE       ,O.H141_TSAMPLENO    AS BARCODE        ,O.H141_SEQNO        AS PID            ,P.A110_CHARTNO      AS CHARTNO        ,P.A110_PATNM        AS PNAME          ,P.A110_JUMIN1       AS AGE            ,P.A110_SEX          AS SEX        FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P   Where P.A110_ChartNo = O.H141_CHARTNO              AND O.H141_ODRDAT BETWEEN '20180605' AND '20180605'   AND O.H141_NOTYYN = 'N'    And O.H141_SUGACD in ('D2800060','D2800020','D2800030') Order By O.H141_ODRDAT, O.H141_SEQNO
