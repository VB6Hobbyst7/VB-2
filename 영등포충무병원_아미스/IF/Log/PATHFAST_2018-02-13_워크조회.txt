SELECT R.SPCMNO as BARCODE, P.PATID as PID, P.PATNAME as PNAME, P.SEX as SEX, O.ACPTDATE as HOSPDATE , O.ACPTSEQ, O.RSVACPTSTATE, O.RESULTSTATE, O.DEPTCODE, O.ORDERDATE, O.SLIPNO, O.IOFLAG, O.ORDERCODE, O.ORDERNAME, R.RESULTFLAG, R.RESULTNO, R.RESULTITEMCODE as ITEM   FROM registinfos O, resultofnum R, PATMST P  WHERE O.acptdate = R.acptdate    AND O.acptdate between '20180213' and '20180213'   AND R.resultitemcode in ('D0040','D4023010','D1073','D4012','D4062010')   AND O.patid = R.patid    AND O.acptseq = R.acptseq    AND O.patid = P.patid    AND R.ORDERCODE IN ('B2640G','D1073','D4012')    AND O.CLAS = 4   ORDER BY R.SPCMNO
