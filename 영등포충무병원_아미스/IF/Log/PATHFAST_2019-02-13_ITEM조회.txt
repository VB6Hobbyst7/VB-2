SELECT R.RESULTITEMCODE as ITEM   FROM registinfos O, resultofnum R  WHERE O.acptdate = R.acptdate    AND R.SPCMNO = '1902130307'   AND R.resultitemcode in ('D4040','D4023010','D4012','D4062010')   AND O.patid = R.patid    AND O.acptseq = R.acptseq    AND O.CLAS = 4    AND R.ORDERCODE IN ('B2640G','D1073','D4012','D4023010','D4040','D4062010')    AND R.RESULTFLAG = 0   ORDER BY R.RESULTITEMCODE
