SELECT R.RESULTITEMCODE as ITEM   FROM registinfos O, resultofnum R  WHERE O.acptdate = R.acptdate    AND R.SPCMNO = '1802130135'   AND R.resultitemcode in ('D0040','D4023010','D1073','D4012','D4062010')   AND O.patid = R.patid    AND O.acptseq = R.acptseq    AND O.CLAS = 4    AND R.ORDERCODE IN ('B2640G','D1073','D4012')    AND R.RESULTFLAG = 0   ORDER BY R.RESULTITEMCODE
