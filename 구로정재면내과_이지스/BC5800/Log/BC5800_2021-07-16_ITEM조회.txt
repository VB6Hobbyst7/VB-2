SELECT DISTINCT a.ORD_CD AS ITEM                   
  FROM H3LAB_RESULT a, H1OPDIN b                   
 WHERE a.ACC_YMD   = '20210716'          
   AND a.RECEPT_NO = '302582'          
   AND a.ORD_CD IN ('B1060','B1091-1-1','B1091-1-10','B1091-1-15','B1091-1-16','B1091-1-4','LEB1050','LYB1010','LYB1020','LYB1040','LYB1050','LYB1060','MCH','MCHC','MCV')            
   AND a.STS_CD     = 'A'                      
   AND (a.RESULT_NM = '' OR a.RESULT_NM IS NULL)
   AND a.RECEPT_NO = b.RECEPT_NO                   
 ORDER BY a.ORD_CD
