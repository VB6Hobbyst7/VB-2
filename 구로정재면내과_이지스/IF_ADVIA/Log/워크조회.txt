SELECT DISTINCT  a.ptnt_no, c.ptnt_nm, a.recept_no, a.spc_cd,  (select codeval1 from pm_mst_div_key1 where codediv = 'LAB01' and codekey1 = a.spc_cd) as spc_nm       , a.sts_cd, a.acc_ymd, a.ord_cd   FROM h3lab_result a, h1opdin b, hz_mst_ptnt c  WHERE a.recept_no = b.recept_no    AND a.sutak_cd = ''   AND a.ptnt_no  = c.ptnt_no   AND a.sts_cd   = 'A'   AND a.acc_ymd between '20200617' AND '20200617'   AND (a.result_nm = '' or a.result_nm is null)   AND a.ord_cd IN ('LEB1050','LYB1050','LYB1040','LYB1010','LYB1020','MCV','MCH','MCHC','B1060','LYB1060','B1091-1','B1091-3','B1091-4')  Order by recept_no
