SELECT DISTINCT   a.RECEIPTDATE AS HOSPDATE, a.SPECIMENNUM AS BARCODE, a.RECEIPTNO AS CHARTNO, a.IPDOPD AS INOUT , a.PTNO AS PID, a.SNAME AS PNAME, a.LABCODE AS ITEM    FROM SLA_LabMaster a,SLA_LabResult b  WHERE a.ORDERCODE IN ('B1034')    AND a.LABCODE IN ('B1034')    AND a.RECEIPTNO = b.RECEIPTNO    AND a.ORDERCODE = b.ORDERCODE    and a.RECEIPTDATE = b.RECEIPTDATE   AND a.SPECIMENNUM = b.SPECIMENNUM   AND a.SPECIMENNUM = '02003396'   AND a.JSTATUS < '3'  ORDER BY a.RECEIPTDATE
