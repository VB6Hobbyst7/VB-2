SELECT DISTINCT   RECEIPTDATE AS HOSPDATE, SPECIMENNUM AS BARCODE, RECEIPTNO AS CHARTNO, IPDOPD AS INOUT , PTNO AS PID, SNAME AS PNAME, COUNT(LABCODE) AS CNT    FROM SLA_LabMaster  WHERE RECEIPTDATE between '2017-06-17' and '2017-06-17'   AND ORDERCODE IN ('Q01')    AND LABCODE IN ('B2570','B2580','B2602','C2200','C2210','C3730','C3750','C3720','C2443','C3780','C3711','B2611','C2420','C2243','C4903','B2710','C2411')    AND JSTATUS < '3'  ORDER BY RECEIPTDATE
