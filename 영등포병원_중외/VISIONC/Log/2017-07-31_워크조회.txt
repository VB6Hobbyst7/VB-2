SELECT DISTINCT   RECEIPTDATE AS HOSPDATE, SPECIMENNUM AS BARCODE, RECEIPTNO AS CHARTNO, IPDOPD AS INOUT , PTNO AS PID, SNAME AS PNAME, COUNT(LABCODE) AS CNT    FROM SLA_LabMaster  WHERE RECEIPTDATE between '2017-07-31' and '2017-07-31'   AND ORDERCODE IN ('B1034')    AND LABCODE IN ('B1034')    AND JSTATUS < '3' GROUP BY RECEIPTDATE, SPECIMENNUM, RECEIPTNO, IPDOPD, PTNO, SNAME  ORDER BY RECEIPTDATE,SPECIMENNUM
