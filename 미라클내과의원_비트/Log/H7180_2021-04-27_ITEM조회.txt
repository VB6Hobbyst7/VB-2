   FROM RESINF AS R                                    
 WHERE LTRIM(RTRIM(R.ResOcmNum)) = '1000000036'
   AND R.RESLABCOD IN ('B/C ratio','B2570','B2580','B2602','B2710','C2200','C2210','C2411','C2420','C2430','C2443','C3711','C3711-1','C3720','C3730','C3750','C3780','LDL-C')             
   AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')     
   AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)
