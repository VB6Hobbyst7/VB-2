 SELECT Distinct R.RESACPNUM,R.RESACPDTE, R.RESMZHTYP, R.RESLABCOD ,SUBSTRING(R.RESACPDTE,1,8) AS 접수일시, R.RESCHTNUM AS 차트번호, P.PBSPATNAM AS 환자명, P.PBSRESNUM AS 주민, R.RESSPMNUM AS 바코드
                ,R.RESACPDTM AS 접수일자, R.RESSHTNAM AS 검사명    FROM RESINF R, PBSINF P 
  WHERE R.RESCHTNUM = P.PBSCHTNUM  
    AND R.RESACPDTE Between '20201119' AND '20201119'
    AND R.RESLABCOD IN ('B2570','B2580','B2590','B2602','B2710','BRC07','C2200','C2210','C2411','C2420','C2443','C3711','C3711B','C3720','C3730','C3750','C3791','C3792','C3793')
   AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F') 
   AND (R.RESMZHMNT = ''  OR R.RESMZHMNT = ' '  OR R.RESMZHMNT IS NULL)
 ORDER BY R.RESACPDTE,R.RESACPNUM, R.RESMZHTYP ASC, R.RESLABCOD ASC
