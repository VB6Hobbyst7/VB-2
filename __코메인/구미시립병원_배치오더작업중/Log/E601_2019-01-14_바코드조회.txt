SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE       ,O.H141_TSAMPLENO     AS BARCODE        ,P.A110_CHARTNO       AS CHARTNO        ,P.A110_PATNM         AS PNAME          ,P.A110_JUMIN1        AS AGE            ,P.A110_SEX           AS SEX            ,O.H141_SUGACD        AS ITEM           ,O.H141_SEQNO         AS SUBITEM    FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P   Where P.A110_ChartNo      = O.H141_CHARTNO         AND O.H141_TSAMPLENO    = '09607740'   AND O.H141_NOTYYN       = 'N'                    And O.H141_SUGACD in ('C3360','C3340','C3290','CZ332','CZ476','C4812','C4802','C4712','C4872')      Order By O.H141_SUGACD
