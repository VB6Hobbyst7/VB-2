SELECT DISTINCT ORD_CD AS ITEM                     
  FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c    
 WHERE a.ACC_YMD   = '20190417'          
   AND a.RECEPT_NO = '104647'          
   AND a.ORD_CD IN ('HBA1C-0','HBA1C-2')            
   AND a.STS_CD    = 'A'                           
   AND a.SUTAK_CD  = ''                            
   AND a.RECEPT_NO = b.RECEPT_NO                   
 ORDER BY ORD_CD
