SELECT DISTINCT        a.ACC_YMD           AS HOSPDATE     
     , a.RECEPT_NO         AS BARCODE      
     , a.PTNT_NO           AS PID          
     , c.PTNT_NM           AS PNAME        
     , c.BIRTH_YMD         AS AGE          
     , c.SEX               AS SEX          
     , a.ORD_CD            AS ITEM         
  FROM H3LAB_RESULT a, H1OPDIN b, HZ_MST_PTNT c 
 WHERE a.RECEPT_NO = '6677'  
   AND a.RECEPT_NO = b.RECEPT_NO           
   AND a.PTNT_NO   = c.PTNT_NO             
   AND a.STS_CD    = 'A'                   
   AND a.SUTAK_CD  = ''                    
   AND a.ORD_CD IN ('HBA1C-0','HBA1C-2')    
 ORDER BY a.ACC_YMD,a.RECEPT_NO,a.ORD_CD
