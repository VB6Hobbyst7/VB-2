 SELECT DISTINCT         R.RESACPDTM                AS HOSPDATE                       , R.RESCHTNUM                AS CHARTNO                        , P.PBSPATNAM                AS PNAME                          , R.RESSPMNUM                AS BARCODE                        , R.RESSHTNAM                AS ITEM                       FROM RESINF R, PBSINF P                                       WHERE R.RESACPDTE Between '20181004' AND '20181005'      AND R.RESCHTNUM    = P.PBSCHTNUM                               AND R.RESLABCOD IN ('B1540','B153001','B1530','B1601','B1723','B1723')                        AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')                AND (R.RESMZHMNT = ''  OR R.RESMZHMNT = ' '  OR R.RESMZHMNT IS NULL) ORDER BY R.RESACPDTM,R.RESCHTNUM ASC
