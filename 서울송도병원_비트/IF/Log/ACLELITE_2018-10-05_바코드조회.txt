 SELECT Distinct         R.RESSPMNUM    AS BARCODE        , R.RESACPDTM    AS HOSPDATE       , R.RESCHTNUM    AS CHARTNO        , P.PBSPATNAM    AS PNAME          , P.PBSRESNUM    AS PID            , R.RESLABCOD    AS ITEM        FROM RESINF R, PBSINF P           WHERE R.RESCHTNUM = P.PBSCHTNUM      AND R.RESSPMNUM = '1802273856'    AND R.RESLABCOD IN ('B1540','B153001','B1530','B1601','B1723','B1723')
