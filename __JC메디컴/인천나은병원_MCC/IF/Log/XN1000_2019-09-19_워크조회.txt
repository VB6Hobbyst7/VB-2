SELECT DISTINCT        READING_YMD                 AS HOSPDATE                 
     , BCODE_NO                    AS BARCODE                  
     , PTNT_NO                     AS PID                      
     , PTNT_NM                     AS PNAME                    
     , AGE                         AS AGE                      
     , SEX                         AS SEX                      
     , IO_GB                       AS INOUT                    
     , COUNT(ORD_CD)               AS COUNT                    
  FROM LIS_INTERFACE1_V                                        
 WHERE READING_YMD BETWEEN '20190918' AND '20190919'   
   AND ORD_CD IN ('LB1050','LB1040','LB1010','LB1020','LH218','LH219','LH220','LB1060','LH10207','LH10208','LH10201','LH10204','LH10205','LH124','LH123','LB1220','LB1230')                          
   AND STS_CD = '0'                                            
 GROUP BY READING_YMD,BCODE_NO,PTNT_NO,PTNT_NM,AGE,SEX,IO_GB   
 ORDER BY READING_YMD,PTNT_NO,BCODE_NO
