SELECT DISTINCT E.slabs_rcode    AS ITEM        FROM BRWONMU..WCHAM A                                                       INNER JOIN OSLABW B     ON A.CHAM_KEY = B.SLABW_CHAM                                           AND A.CHAM_KEY = '0000033300'                                  AND B.slabw_status = '1'                       INNER JOIN OSLABWS C    ON B.SLABW_DATE = C.SLABWS_DATE                                        AND B.slabw_dept = C.slabws_dept                                       AND B.slabw_cnt = C.slabws_cnt                                         AND B.slabw_slab = C.slabws_slab                                       AND C.slabws_date = '20181206'       INNER JOIN OSLABS E     ON C.SLABWS_SCNT = E.SLABS_CNT                                         AND C.slabws_slab = E.slabs_key                                        AND E.slabs_rcode IN ('ALB','ALP','TBIL','DBIL','BUN','CHOL','HDL','CREA','GLU','GOT','GPT','GGT','TP','TG','CRP','UA','AMYL')                               AND E.slabs_use  = 1                     ORDER BY E.slabs_rcode
