SELECT DISTINCT        READING_YMD     AS HOSPDATE         
     , BCODE_NO        AS BARCODE          
     , PTNT_NO         AS PID              
     , PTNT_NM         AS PNAME            
     , AGE             AS AGE              
     , SEX             AS SEX              
     , IO_GB           AS INOUT            
     , ORD_CD          AS ITEM             
     , SP_CD           AS SPCCD            
  FROM LIS_INTERFACE1_V                    
 WHERE BCODE_NO    = '110470'  
   AND ORD_CD IN ('LC3791','LC3792','LC3793')      
 ORDER BY ORD_CD
