SELECT DISTINCT        CONVERT(NVARCHAR(50),M.접수일자,112)    AS HOSPDATE 
     , M.접수번호                              AS PID      
     , M.차트번호                              AS CHARTNO  
     , M.성명                                  AS PNAME    
     , M.성별                                  AS SEX      
     , M.나이                                  AS AGE      
     , M.거래처명                              AS DEPT     
     , E.검사코드                              AS ITEM     
  FROM VW_검사접수 M, VW_검사결과 R, VW_검사코드 E         
 WHERE M.접수일자      = CONVERT(DATE, '20190802')
   AND M.접수번호      = '03214'                
   AND E.학부코드      = 'C1'            
   AND E.검사코드      IN ('1005','1006','1007-1','1007','D2263|01','D1890|01','D3022A|01','D3022|01','D2611|01','D2300|01','1015','D2310|01','D1840|01','1002','D1830|01','D2613|01','5200','D7813|01','CZ246|01')             
   AND ISNULL(R.보고여부, 'N') <> 'Y'                      
   AND (R.결과값 IS NULL OR R.결과값 = '')                 
   AND M.접수일자      = R.접수일자                        
   AND M.접수번호      = R.접수번호                        
   AND R.검사코드      = E.검사코드
