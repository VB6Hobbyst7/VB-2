SELECT DISTINCT       R.접수일                as HOSPDATE      , R.처방번호              as BARCODE       , P.등록번호              as PID           , R.접수번호              as CHARTNO       , P.성명                  as PNAME         , M.외래입원              as INOUT         , COUNT(R.검사코드)       as CNT        FROM DBinfor..내원정보 M, DBinfor..환자정보 P, DBsupport..검사결과 R      WHERE M.연번      = R.처방번호                                               AND M.등록번호  = P.등록번호                                               AND R.처방번호  = R.처방번호                                               AND R.접수일 BETWEEN '20190507' and '20190507'                     AND R.검사코드 IN ('B1050','B1040','B1010','B1020','B1060','MCV','MCH','RDW','MCHC','PDW','MPV','B1091-7','B1091-8','B1091-5')                                     AND (R.결과 = '' OR R.결과 IS NULL)                                      GROUP BY R.접수일, R.처방번호, P.등록번호, R.접수번호, P.성명, M.외래입원
