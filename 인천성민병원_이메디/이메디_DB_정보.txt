--오더조회
SELECT DISTINCT R.RESULTDATE, P.PATID, P.PATNAME, R.SPCMNO
FROM RESULTOFNUM R, REGISTINFOS L, PATMST P
WHERE  R.RESULTDATE =:조회일
AND R.SPCMNO = :검체번호
AND R.ACPTDATE = L.ACPTDATE
AND  R.ACPTSEQ = L.ACPTSEQ
AND  R.PATID = P.PATID
  
--결과업데이트
UPDATE RESULTOFNUM
SET     TEXTRESULTVAL = :결과값,
RESULTINDATE = '날짜',
RESULTINTIME = '시간',
RESULTFLAG = 1
WHERE   SPCMNO = :검체번호
AND     RESULTITEMCODE IN
        (SELECT RESULTITEMCODE FROM LABRSLTITEMBASES
        WHERE   CHANNELNO = :CHANNELNO)
