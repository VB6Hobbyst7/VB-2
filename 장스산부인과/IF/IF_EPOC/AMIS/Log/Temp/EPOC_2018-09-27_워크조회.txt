SELECT DISTINCT       O.ACPTDATE              as HOSPDATE      , R.SPCMNO                as BARCODE       , P.PATID                 as PID           , P.PATNAME               as PNAME         , P.SEX                   as SEX           , COUNT(R.RESULTITEMCODE) as CNT        FROM REGISTINFOS O, RESULTOFNUM R, PATMST P                   WHERE O.ACPTDATE  = R.ACPTDATE                                   AND O.PATID     = R.PATID                                      AND O.ACPTSEQ   = R.ACPTSEQ                                    AND O.PATID     = P.PATID                                      AND O.ACPTDATE BETWEEN '20180927' and '20180927'       AND R.RESULTITEMCODE IN ('XUHM014','XE40502','XUHM0016','XLZ003','XLZ002','XLZ004','XLZ0031','XC0042','XC0035-1','XC0043-1','XC0041-1','XLZ0032','XLZ004-1','XLZ004-2')                   AND O.CLAS          = 4                                        AND R.RESULTFLAG    = 0                                      GROUP BY O.ACPTDATE,R.SPCMNO,P.PATID,P.PATNAME,P.SEX           ORDER BY R.SPCMNO
