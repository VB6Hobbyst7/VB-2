SELECT DISTINCT       M.LABDATE   as HOSPDATE, M.PARTGBN, M.LABSEQ, M.SPECIMENCD     , M.NAME      as PNAME     , M.AGE       as AGE     , M.IDLEFT + M.IDRIGHT AS PJUMIN, M.DEPTCD, M.JUBSUGBN     , M.MEDICALREMARK as BARCODE     , M.RESULTENDYN, R.SUBMCD, R.TESTITEMSEQ as ITEM, R.RESULTDATE, R.RESULTTIME, R.ROUTINECD, R.MACHINECD, R.RESULT, R.DELTAGBN, R.REFGBN, R.PANICGBN, R.DELTAMARK, R.REFMARK, R.PANICMARK, R.CMCODE   FROM H_JUBSU M, H_RESULT R  WHERE M.MEDICALREMARK = 'ISECNT'   AND M.LABDATE       = R.LABDATE    AND M.PARTGBN       = R.PARTGBN    AND M.LABSEQ        = R.LABSEQ    AND M.SPECIMENCD    = R.SPECIMENCD   And R.TESTITEMSEQ   IN ('100','125','126','127')   AND R.CONFIRM       <> 'Y'
