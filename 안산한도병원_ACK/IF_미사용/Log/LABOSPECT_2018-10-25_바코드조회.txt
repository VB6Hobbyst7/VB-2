SELECT DISTINCT       M.LABDATE   as HOSPDATE, M.PARTGBN, M.LABSEQ, M.SPECIMENCD     , M.NAME      as PNAME     , M.REGNO     as PID     , M.AGE       as AGE     , M.SEX       as SEX     , M.IDLEFT + M.IDRIGHT AS PJUMIN, M.DEPTCD, M.JUBSUGBN     , M.MEDICALREMARK as BARCODE     , M.RESULTENDYN, R.SUBMCD, R.TESTITEMSEQ as ITEM, R.RESULTDATE, R.RESULTTIME, R.ROUTINECD, R.MACHINECD, R.RESULT, R.DELTAGBN, R.REFGBN, R.PANICGBN, R.DELTAMARK, R.REFMARK, R.PANICMARK, R.CMCODE   FROM C_JUBSU M, C_RESULT R  WHERE M.MEDICALREMARK = '09406353'   AND M.LABDATE       = R.LABDATE    AND M.PARTGBN       = R.PARTGBN    AND M.LABSEQ        = R.LABSEQ    AND M.SPECIMENCD    = R.SPECIMENCD   And R.TESTITEMSEQ   IN ('001','103','003','028','089','125','092','126','095','127')   AND R.CONFIRM       <> 'Y'
