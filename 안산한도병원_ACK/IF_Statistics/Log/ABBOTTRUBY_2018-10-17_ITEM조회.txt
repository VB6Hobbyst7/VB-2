SELECT DISTINCT L.LABODRCOD as ITEM                  FROM ME_LABDAT L, ME_DAT D                        WHERE L.LABCHTNUM  = ''            AND L.LABODRDTE  = ''            AND L.LABKEYNUM  = D.DATKEYNUM                     AND L.LABATTEND  = D.DATATTEND                     AND L.LABCHTNUM  = D.DATCHTNUM                     AND L.LABCHTNUM  = M.MANCHTNUM                     AND L.LABODRDTE  = D.DATODRDTE                     AND L.LABODRCOD IN ('B1050002','B1040002','B1010002','B1020002','B1060002','B1010A','B1010B','PCT','B1010C','B1060B','B1091C','B1091A','B1050-3','B1050-4','B1050-5','B1060D')            AND (L.LABCANCEL = '' OR L.LABCANCEL IS NULL)      AND (L.LABRESULT = ''  OR L.LABRESULT IS NULL)     AND L.LABENDDEP < '3'                            Order By L.LABODRCOD
