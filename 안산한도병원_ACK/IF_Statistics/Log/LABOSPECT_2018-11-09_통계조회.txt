select r.labdate as stadate,r.testitemseq,t.printnm,count(r.result) as cnt  from c_result r, testitem t where r.labdate between '20181107' and '20181109'   and r.machinecd = '!'   and r.result <> ''   and r.partgbn = '01'   and r.SPECIMENCD = '002'   and t.partcd = 'C'   and r.partgbn = t.partgbn   and r.SPECIMENCD = t.SPECIMENCD   and r.testitemseq = t.testitemseq group by r.labdate, r.testitemseq,t.printnm order by r.labdate, r.testitemseq,t.printnm
