SELECT DISTINCT RECEIPTDATE as 접수일자, SPECIMENNUM as 바코드번호, RECEIPTNO as 챠트번호, IPDOPD, PTNO as 내원번호, SNAME as 이름, LABCODE as ITEM,ORDERCODE
  FROM SLA_LabMaster 
 WHERE RECEIPTDATE between '2021-03-03' and '2021-03-03'
   AND LABCODE IN ('B1050','B1040','B1010','B1020','B1060-1','B1060-2','B1060-3','B1060-4','B1091-2','B1091-3','B1091-1','B1220','B1220','B1330','LCBC-6','P-LCR','LCBC-7') 
   AND JSTATUS < '3'
  ORDER BY RECEIPTDATE
