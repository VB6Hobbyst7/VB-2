  FROM SLA_LabMaster 
 WHERE RECEIPTDATE between '2019-05-22' and '2019-05-22'
   AND LABCODE IN ('D1840-','D1880-','D3022-5','D1860-','D1850-','D1870-','D2611-','D2613-','D2300-','D2280-','D1830-','D1890-','D2263-','G2','G7','G8','G3','G4','G10','G9','G5','D3022-6','D3022','D3022-1','D3022-2') 
   AND JSTATUS < '3'
  ORDER BY RECEIPTDATE,SPECIMENNUM
