SELECT DISTINCT RECEIPTDATE as 접수일자, SPECIMENNUM as 바코드번호, RECEIPTNO as 챠트번호, IPDOPD, PTNO as 내원번호, SNAME as 이름, LABCODE as ITEM
  FROM SLA_LabMaster 
 WHERE RECEIPTDATE between '2016-07-06' and '2016-07-06'
   AND LABCODE IN ('B003001','B003002','B003005','B003004','B003010','B003006','B003007','B003008','B0030009','B003011') 
   AND JSTATUS < '3'
  ORDER BY RECEIPTDATE
