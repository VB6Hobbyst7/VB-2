SELECT DISTINCT       R.접수일                as HOSPDATE      , R.처방번호              as BARCODE        , P.등록번호              as PID            , R.접수번호              as CHARTNO        , P.성명                  as PNAME          , M.외래입원              as INOUT          , R.검사코드              as ITEM           , R.순번                  as ITEMSEQ     FROM DBinfor..내원정보 M, DBinfor..환자정보 P, DBsupport..검사결과 R      WHERE M.연번      = R.처방번호                                               AND M.등록번호  = P.등록번호                                               AND R.처방번호  = R.처방번호                                               AND R.처방번호 = '1'     AND R.검사코드 IN ('D2253')
