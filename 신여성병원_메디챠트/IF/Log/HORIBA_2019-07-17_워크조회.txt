SELECT DISTINCT        L.LABODRDTE     as      HOSPDATE    
     , L.LABBARCOD     as      BARCODE     
     , L.LABCHTNUM     as      CHARTNO     
     , L.LABATTEND     as      PID         
     , M.MANADMFOR     as      INOUT       
     , M.MANRESNUM     as      JUMIN       
     , M.MANPATNAM     as      PNAME       
     , L.LABODRSTP     as      ORDCODE     
  FROM ME_LABDAT L, ME_DAT D, ME_MAN M     
 WHERE L.LABODRDTE between  '20190717' AND '20190717235959'
   AND L.LABKEYNUM = D.DATKEYNUM                   
   AND L.LABATTEND = D.DATATTEND                   
   AND L.LABATTEND = M.MANATTEND                   
   AND L.LABCHTNUM = D.DATCHTNUM                   
   AND L.LABCHTNUM = M.MANCHTNUM                   
   AND L.LABODRDTE = D.DATODRDTE                   
   AND L.LABODRCOD IN ('B1050','B1040','B1010','B1020','MCV','MCH','MCHC','RDW','B1060','B1091-7','B1091-8','B1091-5')         
   AND (L.LABCANCEL = '' OR L.LABCANCEL IS NULL)   
   AND (L.LABRESULT = ''  OR L.LABRESULT IS NULL)  
   AND L.LABENDDEP < '3'                           
 ORDER BY L.LABODRDTE, L.LABCHTNUM, L.LABBARCOD, L.LABINSNUM
