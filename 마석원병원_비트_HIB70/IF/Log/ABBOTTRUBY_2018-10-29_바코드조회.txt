SELECT DISTINCT        L.LABSERIAL                   ,L.LABODRDTE as HOSPDATE       ,L.LABBARCOD as BARCODE        ,L.LABATTEND as PID            ,L.LABCHTNUM as CHARTNO        ,M.MANADMFOR as IO             ,M.MANRESNUM as JUMIN          ,M.MANPATNAM as PNAME          ,L.LABODRCOD as ITEM           ,L.LABODRSTP as SEQ        FROM ME_LABDAT L, ME_DAT D, ME_MAN M              WHERE L.LABCHTNUM = ''             AND L.LABODRDTE = ''             AND L.LABKEYNUM = D.DATKEYNUM                      AND L.LABODRDTE = D.DATODRDTE                      AND L.LABATTEND = D.DATATTEND                      AND L.LABATTEND = M.MANATTEND                      AND L.LABCHTNUM = D.DATCHTNUM                      AND L.LABCHTNUM = M.MANCHTNUM                      AND (L.LABCANCEL = '' OR L.LABCANCEL IS NULL)      AND (L.LABRESULT = ''  OR L.LABRESULT IS NULL)     AND L.LABODRCOD IN ('B1050','B1040','B1010','B1020','C1060A','C1060B','C1060C','B1220','BXXXX','C1060D','CXXXX','CXXXX','B1091A','B1091B','B1091C','BXXXX','BXXXX')            AND L.LABENDDEP < '3'
