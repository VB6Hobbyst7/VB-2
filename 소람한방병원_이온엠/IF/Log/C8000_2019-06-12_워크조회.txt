SELECT DISTINCT        O.H141_ODRDAT        AS HOSPDATE 
      ,O.H141_TSAMPLENO     AS BARCODE  
      ,O.H141_SEQNO         AS PID      
      ,P.A110_CHARTNO       AS CHARTNO  
      ,P.A110_PATNM         AS PNAME    
      ,P.A110_JUMIN1        AS AGE      
      ,P.A110_SEX           AS SEX      
      ,COUNT(O.H141_SUGACD) AS CNT      
  FROM TB_H141_LISTAKEBODY O, TB_A110_PATINFO P                
 Where O.H141_ODRDAT BETWEEN '20190612' AND '20190612' 
   AND P.A110_ChartNo = O.H141_CHARTNO                         
   AND O.H141_NOTYYN       IN ('N','T')                        
   And O.H141_SUGACD IN ('SO-LTSH','SO-LFT4','SO-LFT3','SO-LVITD','SO-LCA199','SO-LSCC','SO-LCA125','SO-LCYFRA','SO-LCEA','SO-LCA153')                   
 Group By O.H141_ODRDAT,O.H141_TSAMPLENO,O.H141_SEQNO,P.A110_CHARTNO,P.A110_PATNM,P.A110_JUMIN1,P.A110_SEX 
 Order By O.H141_ODRDAT, O.H141_SEQNO
