Select DISTINCT a.ORDT as HOSPDATE, b.PANM as PNAME, a.SPNO as BARCODE, a.PAID as PID, a.OIFL as INOUT, b.SEXS as SEX, b.AGES as AGE, a.NWNO as CHARTNO, a.ORCD as ITEM   From emr.LRESULT a, emr.APATINF b Where a.SPNO = '201804030032'   And a.PAID = b.PAID    And a.ORCD in ('C3792','C3791','C3793')   And a.OKFL = 'N'    AND (a.RSFL IS NULL OR a.RSFL = 'N' OR a.RSFL = '') ORDER BY a.ORDT, a.SPNO
