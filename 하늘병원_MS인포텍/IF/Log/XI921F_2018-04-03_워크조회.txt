Select DISTINCT a.ORDT as HOSPDATE, b.PANM as PNAME,a.SPNO as BARCODE,a.PAID as PID, a.OIFL as INOUT,b.SEXS as SEX,b.AGES as AGE,a.NWNO as CHARTNO  From emr.LRESULT a, emr.APATINF b Where a.ORDT between  '20180403' and '20180403'   And a.PAID = b.PAID    And a.ORCD in ('C3791','C3792','C3793')   And a.OKFL = 'N'    AND (a.RSFL IS NULL OR a.RSFL = 'N' OR a.RSFL = '') ORDER BY a.ORDT, a.SPNO
