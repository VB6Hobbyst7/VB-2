Select DISTINCT a.ORDT as HOSPDATE, b.PANM as PNAME,a.SPNO as BARCODE,a.PAID as PID, a.OIFL as INOUT,b.SEXS as SEX,b.AGES as AGE,a.NWNO as CHARTNO  From emr.LRESULT a, emr.APATINF b Where a.ORDT between  '20180313' and '20180313'   And a.PAID = b.PAID    And a.ORCD in ('C2200','C2210','C3720','C3730','C3750','B2570','B2580','B2710','B2602','C2411','C2443','C2420','C3711','C3795','C3794','C2243','C4903','C3780','CZ241-1','CZ241-2','CZ24','CZ241','B2590')   And a.OKFL = 'N'    AND (a.RSFL IS NULL OR a.RSFL = 'N' OR a.RSFL = '') ORDER BY a.ORDT, a.SPNO
