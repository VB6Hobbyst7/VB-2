SELECT L.LABSERIAL, L.LABATTEND as 내원번호, L.LABCHTNUM as 챠트번호, L.LABODRDTE as 접수일자, M.MANADMFOR as 입외,
       M.MANRESNUM as 주민번호, M.MANPATNAM as 이름, L.LABINSNUM as 처방순서,L.LABSMPNAM as 검체명, L.LABBARCOD as 바코드번호, L.LABODRCOD as ITEM, L.LABODRSTP as SEQ 
  FROM ME_LABDAT L, ME_DAT D, ME_MAN M
 WHERE L.LABODRDTE between  '2016-11-18' AND '2016-11-18'
   AND L.LABKEYNUM = D.DATKEYNUM 
   AND L.LABATTEND = D.DATATTEND 
   AND L.LABATTEND = M.MANATTEND 
   AND L.LABCHTNUM = D.DATCHTNUM 
   AND L.LABCHTNUM = M.MANCHTNUM 
   AND L.LABODRDTE = D.DATODRDTE 
   AND L.LABODRCOD IN ('C2200','C2210','B2570','B2580','B2710','B2590','B2630','B2602','C2411','C2420','CZ133','C2443','C3711','C3730','C3750','C3780','C2243','C2241','C4901','C4903','B2611','C3794','C','C3797','C3795','B2590','C2490','C2510','B2621','C3720','C','C3791','C3792','C3793')
   AND (L.LABCANCEL = '' OR L.LABCANCEL IS NULL) 
   AND (L.LABRESULT = ''  OR L.LABRESULT IS NULL)
   AND L.LABENDDEP < '3' 
 ORDER BY L.LABODRDTE, L.LABBARCOD, L.LABINSNUM, L.LABODRCOD
