Select M.PART_JUBSU_DATE   AS HOSPDATE              , M.PART_JUBSU_TIME                            , M.BUNHO             AS PID                   , M.SUNAME            AS PNAME                 , M.SPECIMEN_SER      AS BARCODE               , M.AGE               AS AGE                   , M.SEX               AS SEX                   , M.GWA_NAME          AS INOUT                 , M.HANGMOG_CODE      AS ITEM                  , R.LAB_NO            AS CHARTNO            From MEDI.CPL3020 R                               , MEDI.CPL2010 M                               , MEDI.CPL0101 E                           Where M.PART_JUBSU_DATE Between to_date('20190403','yyyymmdd')                                and to_date('20190410','yyyymmdd')        and NVL(R.CONFIRM_YN, 'N') = 'N'                   and R.JANGBI_CODE   = 'I2'       and E.JANGBI_OUT_CODE is Not Null                  and R.SPECIMEN_SER  = M.SPECIMEN_SER               and R.HANGMOG_CODE  = M.HANGMOG_CODE               and R.SPECIMEN_CODE = M.SPECIMEN_CODE              and R.HANGMOG_CODE  = E.HANGMOG_CODE               and R.SPECIMEN_CODE = E.SPECIMEN_CODE
