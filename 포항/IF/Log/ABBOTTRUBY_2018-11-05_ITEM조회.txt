SELECT DISTINCT L.LABODRCOD as ITEM                  FROM ME_LABDAT L, ME_DAT D                        WHERE L.LABCHTNUM  = '00120856'            AND L.LABODRDTE  = '2018-11-05'            AND L.LABKEYNUM  = D.DATKEYNUM                     AND L.LABATTEND  = D.DATATTEND                     AND L.LABCHTNUM  = D.DATCHTNUM                     AND L.LABODRDTE  = D.DATODRDTE                     AND L.LABODRCOD IN ('B1050','B1040','B1010','B1020','C1060A','C1060B','C1060C','B1220','B1060','C1060D','C1060E','D0002060','B1091A','B1091B','B1091C','B1091D','B1091E')            AND (L.LABCANCEL = '' OR L.LABCANCEL IS NULL)      AND (L.LABRESULT = ''  OR L.LABRESULT IS NULL)     AND L.LABENDDEP < '3'                            Order By L.LABODRCOD
