SELECT DiSTINCT b.SCP42JDATE as 접수일자, a.SCP41SPMNO2 as 바코드번호, b.SCP42IDNOA as 내원번호, a.SCP41NAME as 이름, a.SCP41SEX as 성별, a.SCP41BIRTH as 나이,b.SCP42SUGACD as ITEM
  FROM JAIN_SCP.SCPRST41 a, JAIN_SCP.SCPRST42 b 
 WHERE a.SCP41PCODE = b.SCP42PCODE
   AND a.SCP41JDATE = b.SCP42JDATE
   AND a.SCP41SID   = b.SCP42SID
   AND a.SCP41SPMNO2 = b.SCP42SPMNO2 
   AND a.SCP41SPMNO2 = 'zx'
   AND (b.SCP42RESULT IS NULL OR b.SCP42RESULT = '')
