SELECT DISTINCT RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) AS ORDCD  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND a.APATMRNO = '00003216'   AND b.OBODCASE = '내원번호'   AND b.OBODORNO = '오더번호'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('R697|','FERR|','33BL|','K-AB|','24AP|','K-AG|','24AN1|','33BC|','HCV|','R770|','33AN6|','R750|','R504|','R667|','R501|','PSA|','AFPKG1|','R986|','AFP|','AFPKG|','R838|','CA19-9|','33BA|','CEA|')    AND b.OBODSTAT = 'AC'  Order by ORDCD
