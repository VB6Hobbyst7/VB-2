SELECT DISTINCT RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) AS ORDCD  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND a.APATMRNO = '00000022'   AND b.OBODCASE = '224974'   AND b.OBODORNO = '9'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('21AC|','21AD|002','21AD|001','21AE|001','21AF|','21AF1|002','21AF1|001','21AF1|003','21AF1|004','21AK|','B1091','B1091','B1091','C3815|E','C3815|F','C3815|H')    AND b.OBODSTAT = 'AC'  Order by ORDCD
