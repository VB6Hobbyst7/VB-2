SELECT DISTINCT b.OBODORDT, a.APATMRNO, b.OBODCASE, b.OBODORNO, b.OBODORSQ, b.OBODIOGB, a.APATNAME, a.APATPSEX, a.APATJMN1, b.OBODCODE, c.OBSUCODE, c.OBSUSUBC  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND b.OBODORDT between '20170227' AND '20170228'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('21AC|','21AD|002','21AD|001','21AE|001','21AF|','21AF1|002','21AF1|001','21AF1|003','21AF1|004','21AK|','B1091','B1091','B1091','C3815|E','C3815|F','C3815|H')    AND b.OBODSTAT = 'AC'  Order by b.OBODORDT,a.APATMRNO,b.OBODORNO,b.OBODORSQ
