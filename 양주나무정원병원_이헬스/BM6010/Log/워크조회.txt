SELECT DISTINCT b.OBODORDT, a.APATMRNO, b.OBODCASE, b.OBODORNO, b.OBODORSQ, b.OBODIOGB, a.APATNAME, a.APATPSEX, a.APATJMN1, b.OBODCODE, c.OBSUCODE, c.OBSUSUBC  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND b.OBODORDT between '20170324' AND '20170325'   AND RTRIM(LTRIM(c.OBSUCODE)) + '|' + RTRIM(LTRIM(c.OBSUSUBC)) IN ('23AM|','23AN|','23AK|','23AP|','23AR|','23AD|','23BK|','23AE|','23BB|','23AF|','23AH|','23AI|','23AJ|001','23AG|','23BI|','24AB|','23BC|','C2510|')    AND b.OBODSTAT = 'AC'  Order by b.OBODORDT,a.APATMRNO,b.OBODORNO,b.OBODORSQ
