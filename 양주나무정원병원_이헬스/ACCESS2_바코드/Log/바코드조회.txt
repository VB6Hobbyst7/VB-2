SELECT DISTINCT RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) AS ORDCD  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND a.APATMRNO = '00000022'   AND b.OBODCASE = '224974'   AND b.OBODORNO = '12'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('23AB|','23AC|','23AD|','23AE|','23AF|','23AH|','23AG|','23AI|','23AJ|','23AK|','23AL|','23AM|','23AN|','23AP|','23AR|','23AT|','23AU|','23AV|','23AW|','23AY|')    AND b.OBODSTAT = 'AC'  Order by ORDCD
