SELECT DISTINCT b.OBODORDT, a.APATMRNO, b.OBODCASE, b.OBODORNO, b.OBODORSQ, b.OBODIOGB, a.APATNAME, a.APATPSEX, a.APATJMN1, b.OBODCODE, c.OBSUCODE, c.OBSUSUBC  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND b.OBODORDT between '20170203' AND '20170228'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('22AD|','22AH|','22AE|','22AGN|','22AV|','22AX|','22AF|','22AJ|','22AK|','22AI|')    AND b.OBODSTAT = 'AC'  Order by b.OBODORDT,a.APATMRNO,b.OBODORNO,b.OBODORSQ
