SELECT DISTINCT RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) AS ORDCD  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND a.APATMRNO = '00000022'   AND b.OBODCASE = '224974'   AND b.OBODORNO = '4'   AND RTRIM(c.OBSUCODE) + '|' + RTRIM(c.OBSUSUBC) IN ('22AD|','22AH|','22AE|','22AGN|','22AV|','22AX|','22AF|','22AJ|','22AK|','22AI|')    AND b.OBODSTAT = 'AC'  Order by ORDCD
