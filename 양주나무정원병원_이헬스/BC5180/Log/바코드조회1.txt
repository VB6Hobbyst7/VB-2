SELECT DISTINCT b.OBODORDT, a.APATMRNO, b.OBODCASE, b.OBODORNO, b.OBODORSQ, b.OBODIOGB, a.APATNAME, a.APATPSEX, a.APATJMN1, b.OBODCODE, c.OBSUCODE, c.OBSUSUBC  FROM ABPATMST a      ,OBODRMTM b      ,OBSURSTM c  WHERE a.APATMRNO = b.OBODMRNO    AND a.APATMRNO = c.OBSUMRNO    AND b.OBODCASE = c.OBSUCASE    AND b.OBODORNO = c.OBSUORNO    AND b.OBODORSQ = c.OBSUORSQ    AND (c.OBSURSLT IS NULL OR c.OBSURSLT = '')   AND RTRIM(LTRIM(c.OBSUCODE)) + '|' + RTRIM(LTRIM(c.OBSUSUBC)) IN ('21AC|','21AD|002','21AD|001','21AE|001','21AF|002','21AF|001','21AF1|001','21AF1|002','21AF1|003','21AF1|004','21AF|','21AK|','21AN|001','21AN|002','21AN|003','21AN|004','21AN|005','21AN|006')    AND b.OBODSTAT = 'AC'    AND a.APATMRNO = '12345678' Order by b.OBODORDT,a.APATMRNO,b.OBODORNO,b.OBODORSQ
