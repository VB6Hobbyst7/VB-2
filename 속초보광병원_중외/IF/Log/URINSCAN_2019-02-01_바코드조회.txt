SELECT DISTINCT        a.RECEIPTDATE    AS HOSPDATE      , a.SPECIMENNUM    AS BARCODE       , a.RECEIPTNO      AS CHARTNO       , a.IPDOPD         AS INOUT         , a.PTNO           AS PID           , a.SNAME          AS PNAME         , a.ORDERCODE      AS ORDCODE       , b.LABCODE        AS ITEM        FROM SLA_LabMaster a, SLA_LabResult b    WHERE a.RECEIPTNO     = b.RECEIPTNO          AND a.ORDERCODE     = b.ORDERCODE          and a.RECEIPTDATE   = b.RECEIPTDATE        AND a.SPECIMENNUM   = b.SPECIMENNUM        AND a.SPECIMENNUM   = '02228944'   AND b.LABCODE IN ('B0010G','B0010H','B0010A','B0010I','B0010C','B0010B','B0010F','B0010D','B0010E','B0010J','COLOR')      AND a.JSTATUS < '3'                      ORDER BY b.LABCODE
