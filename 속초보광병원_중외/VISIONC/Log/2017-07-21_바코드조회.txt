SELECT DISTINCT   RECEIPTDATE AS HOSPDATE, SPECIMENNUM AS BARCODE, RECEIPTNO AS CHARTNO, IPDOPD AS INOUT , PTNO AS PID, SNAME AS PNAME, LABCODE AS ITEM , COUNT(LABCODE) AS CNT    FROM SLA_LabMaster  WHERE ORDERCODE IN ('Q01')    AND LABCODE IN ('B2602')    AND SPECIMENNUM = '4309'   AND JSTATUS < '3'  ORDER BY RECEIPTDATE
