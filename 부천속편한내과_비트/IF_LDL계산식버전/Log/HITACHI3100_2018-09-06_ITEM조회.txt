 SELECT DISTINCT R.RESLABCOD AS ITEM                      FROM DRBITPACK..RESINF AS R                          WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '427913'   AND R.RESLABCOD IN ('L6','L7','L5','L8','L1','L0','L2','C3750','l11','l10','C2411','C2443','L16','C2420','L15','C3711','L9','L15','L15','44','c2430','C2430-1')                AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')        AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)       Order By R.RESLABCOD
