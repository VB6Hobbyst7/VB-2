 SELECT DISTINCT R.RESLABCOD AS ITEM                      FROM DRBITPACK..RESINF AS R                          WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '428639'   AND R.RESLABCOD IN ('L22','L23','B1010','L24','L25','B1020A','B1020B','B1020C','B1220','L26','B1060A','B1230','B1060B','B1091A','B1091B','B1091C')                AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')        AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)       Order By R.RESLABCOD
