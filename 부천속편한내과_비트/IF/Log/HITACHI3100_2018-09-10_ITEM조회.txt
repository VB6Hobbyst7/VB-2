 SELECT DISTINCT R.RESLABCOD AS ITEM                      FROM DRBITPACK..RESINF AS R                          WHERE LTRIM(RTRIM(R.RESOCMNUM)) = '428564'   AND R.RESLABCOD IN ('B2570','L6','B2580','L5','L5','L2','B2710','L8','L1','L0','L10','C3750','L11','C3711','L9','C2411','C2443','L16','C2420','L15','c2430','C2430-1')                AND (R.RESREPTYP IS NULL OR R.RESREPTYP <> 'F')        AND (R.RESRLTVAL = ''  OR R.RESRLTVAL IS NULL)       Order By R.RESLABCOD
