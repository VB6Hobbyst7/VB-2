select distinct m.workarea                  , m.accdt AS HOSPDATE                  , m.accseq                             , m.spcyy                              , m.spcno                              , m.ptid AS PID                        , p.수진자명 AS PNAME                  , m.rcvdt                              , m.rcvtm                              , COUNT(r.testcd) AS CNT            from plis..s2lab201 m                      , medichart..TB_인적사항 p                  , plis..s2lab302 r                      , plis..s2lab001 e                  where SUBSTRING(m.accdt,1,8) BETWEEN '20171222' AND '20171222'   and m.workarea = '01'     and r.testcd IN ('B0030-1','B0030-2','B0030-3','B0030-4','B0030-5','B0030-6','B0030-7','B0030-8','B0030-9','B0030-10','B0041001','B0041002','B0041003','B0041004')       and (r.vfydt IS NULL OR r.vfydt='')      and m.ptid = p.챠트번호                    and m.workarea = r.workarea                and m.accdt = r.accdt                      and m.accseq = r.accseq                    and r.testcd = e.testcd                   Group by m.workarea, m.accdt, m.spcyy,m.spcno,m.accseq, m.ptid,p.수진자명,m.rcvdt, m.rcvtm   Order by m.rcvdt, m.rcvtm
