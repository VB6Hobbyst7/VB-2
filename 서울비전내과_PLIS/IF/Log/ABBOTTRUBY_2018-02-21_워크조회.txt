select distinct m.workarea                  , m.accdt AS HOSPDATE                  , m.accseq                             , m.spcyy                              , m.spcno                              , m.ptid AS PID                        , p.수진자명 AS PNAME                  , m.rcvdt                              , m.rcvtm                              , COUNT(r.testcd) AS CNT            from plis..s2lab201 m                      , medichart..TB_인적사항 p                  , plis..s2lab302 r                      , plis..s2lab001 e                  where SUBSTRING(m.accdt,1,8) BETWEEN '20180121' AND '20180221'   and r.testcd IN ('B1050002','B1040002','B1010002','B1020002','B1060002','B1010A','B1010B','PCT','B1010C','B1060B','B1091C','B1091A','B1050-3','B1050-4','B1050-5','B1060D')       and (r.vfydt IS NULL OR r.vfydt='')      and m.ptid = p.챠트번호                    and m.workarea = r.workarea                and m.accdt = r.accdt                      and m.accseq = r.accseq                    and r.testcd = e.testcd                   Group by m.workarea, m.accdt, m.spcyy,m.spcno,m.accseq, m.ptid,p.수진자명,m.rcvdt, m.rcvtm   Order by m.rcvdt, m.rcvtm
